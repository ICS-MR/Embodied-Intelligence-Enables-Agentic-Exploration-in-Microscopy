fiji_initialize()
brightfield_file = "brightfield_zstack.ome.tif"
dapi_file = "dapi_zstack.ome.tif"
fitc_file = "fitc_zstack.ome.tif"
tritc_file = "tritc_zstack.ome.tif"

say("[ACTION] Loading brightfield Z-stack image: " + brightfield_file)
brightfield_img = load_image(brightfield_file)
say("[ACTION] Loading DAPI Z-stack image: " + dapi_file)
dapi_img = load_image(dapi_file)
say("[ACTION] Loading FITC Z-stack image: " + fitc_file)
fitc_img = load_image(fitc_file)
say("[ACTION] Loading TRITC Z-stack image: " + tritc_file)
tritc_img = load_image(tritc_file)

say("[ACTION] Performing extended depth of field processing on brightfield Z-stack")
brightfield_proj = z_projection(brightfield_img, method="max")
say("[ACTION] Performing extended depth of field processing on DAPI Z-stack")
dapi_proj = z_projection(dapi_img, method="max")
say("[ACTION] Performing extended depth of field processing on FITC Z-stack")
fitc_proj = z_projection(fitc_img, method="max")
say("[ACTION] Performing extended depth of field processing on TRITC Z-stack")
tritc_proj = z_projection(tritc_img, method="max")

say("[ACTION] Saving single-plane images after extended depth of field processing")
save_image(brightfield_proj, "brightfield_edf.ome.tif", "Brightfield single-plane image after extended depth of field")
save_image(dapi_proj, "dapi_edf.ome.tif", "DAPI single-plane image after extended depth of field")
save_image(fitc_proj, "fitc_edf.ome.tif", "FITC single-plane image after extended depth of field")
save_image(tritc_proj, "tritc_edf.ome.tif", "TRITC single-plane image after extended depth of field")

say("[ACTION] Merging brightfield, DAPI, FITC, and TRITC images into composite image")
channels_to_merge = [brightfield_proj, dapi_proj, fitc_proj, tritc_proj]
channel_colors = ["Gray", "Blue", "Green", "Red"]
output_file = "composite_brightfield_dapi_fitc_tritc.tif"
merged_image = merge_channels(channels_to_merge, colors=channel_colors, outpath=output_file)

say("[ACTION] Saving composite overlay image to: " + output_file)
save_image(merged_image, output_file, "Composite overlay of brightfield, DAPI, FITC, and TRITC after extended depth of field")
fiji_shutdown()
