```python
fiji_initialize()
input_file = "brightfield_3x3mm.ome.tif"
say("[ACTION] Loading brightfield image of 3×3mm region: " + input_file)
image = load_image(input_file)
say("[ACTION] Converting image to numpy array for annotation")
img_np = convert_to_numpy(image)
if img_np.ndim == 3:
    img_np = img_np[0]
height, width = img_np.shape
pixel_size_um = 0.32
scalebar_length_um = 100
scalebar_length_px = int(scalebar_length_um / pixel_size_um)
scalebar_height = max(4, height // 100)
scalebar_margin = 20
scalebar_color = (255, 255, 255)
text = "100 μm"
font = cv.FONT_HERSHEY_SIMPLEX
font_scale = 0.7
font_thickness = 2
text_color = (255, 255, 255)
if len(img_np.shape) == 2:
    img_rgb = cv.cvtColor(img_np, cv.COLOR_GRAY2BGR)
else:
    img_rgb = img_np.copy()
scalebar_x1 = scalebar_margin
scalebar_y1 = height - scalebar_margin - scalebar_height
scalebar_x2 = scalebar_margin + scalebar_length_px
scalebar_y2 = height - scalebar_margin
cv.rectangle(img_rgb, (scalebar_x1, scalebar_y1), (scalebar_x2, scalebar_y2), scalebar_color, -1)
(text_width, text_height), baseline = cv.getTextSize(text, font, font_scale, font_thickness)
text_x = scalebar_x1
text_y = scalebar_y1 - 10
if text_y - text_height < 0:
    text_y = scalebar_y1 + text_height + 10
cv.putText(img_rgb, text, (text_x, text_y), font, font_scale, text_color, font_thickness, cv.LINE_AA)
say("[ACTION] Displaying annotated image for 10 seconds")
plt.imshow(cv.cvtColor(img_rgb, cv.COLOR_BGR2RGB))
plt.axis('off')
plt.show(block=False)
plt.pause(10)
plt.close()
fiji_shutdown()
```
