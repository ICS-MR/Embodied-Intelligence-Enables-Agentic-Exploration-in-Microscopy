```python
fiji_initialize()
input_file = "brightfield_3mm_region.ome.tif"
say("[ACTION] Loading brightfield image of 3 mm × 3 mm region: " + input_file)
image = load_image(input_file)
say("[ACTION] Converting image to numpy array for annotation")
img_np = convert_to_numpy(image)
if img_np.ndim == 3:
    img_np = img_np[0]
height, width = img_np.shape
pixel_size_um = 0.32
scalebar_length_um = 1000
scalebar_length_px = int(scalebar_length_um / pixel_size_um)
scalebar_height_px = max(5, int(0.01 * height))
scalebar_color = (255, 255, 255)
if img_np.dtype != np.uint8:
    img_np = cv.normalize(img_np, None, 0, 255, cv.NORM_MINMAX).astype(np.uint8)
img_rgb = cv.cvtColor(img_np, cv.COLOR_GRAY2RGB)
margin = 30
start_x = margin
start_y = height - margin
end_x = start_x + scalebar_length_px
end_y = start_y
cv.line(img_rgb, (start_x, start_y), (end_x, end_y), scalebar_color, scalebar_height_px)
font = cv.FONT_HERSHEY_SIMPLEX
font_scale = 0.8
font_thickness = 2
label_text = "1 mm"
(text_width, text_height), _ = cv.getTextSize(label_text, font, font_scale, font_thickness)
text_x = start_x
text_y = start_y - 10
cv.putText(img_rgb, label_text, (text_x, text_y), font, font_scale, scalebar_color, font_thickness, cv.LINE_AA)
corner_label = "3 mm × 3 mm region"
corner_text_x = margin
corner_text_y = height - margin - scalebar_height_px - text_height - 20
cv.putText(img_rgb, corner_label, (corner_text_x, corner_text_y), font, font_scale, scalebar_color, font_thickness, cv.LINE_AA)
say("[ACTION] Displaying annotated image for 10 seconds")
plt.imshow(img_rgb)
plt.axis('off')
plt.show(block=False)
plt.pause(10)
plt.close()
fiji_shutdown()
```
