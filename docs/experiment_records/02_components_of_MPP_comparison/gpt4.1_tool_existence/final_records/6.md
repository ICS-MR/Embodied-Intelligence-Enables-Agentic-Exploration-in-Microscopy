```python
say("[ACTION] Generating 24-well plate positions")
positions = create_24_wells_positions()

say("[ACTION] Setting objective lens to 20x")
set_objective('3-LUCPLFLN20XRC')

say("[ACTION] Setting filter set to DAPI (blue fluorescence)")
set_channel('2-U-FUNA')

say("[ACTION] Setting exposure time for DAPI")
set_exposure(400)
exposure_time = get_exposure()
if exposure_time < 200 or exposure_time > 1000:
    exposure_time = 400
    set_exposure(exposure_time)

say("[ACTION] Setting halogen lamp brightness to 0 for fluorescence imaging")
set_brightness(0)

for idx, (x, y) in enumerate(positions):
    pos_name = f"well_{idx+1}"
    say(f"[ACTION] Setting XY position for {pos_name}")
    if x < 0 or x > 500000 or y < 0 or y > 500000:
        say(f"[ERROR] Position out of range for {pos_name}: ({x}, {y})")
        continue
    set_x_y_position(x, y)
    say(f"[ACTION] Autofocusing at {pos_name} in DAPI mode")
    optimal_z = perform_autofocus()
    if optimal_z < 0 or optimal_z > 10000:
        say(f"[ERROR] Autofocus Z out of range for {pos_name}: {optimal_z}")
        continue
    set_z_position(optimal_z)
    say(f"[ACTION] Analyzing Z-stack range for {pos_name}")
    z_max, z_min = z_stack_range()
    if z_min < 0: z_min = 0
    if z_max > 10000: z_max = 10000
    if z_max <= z_min:
        say(f"[ERROR] Invalid Z-stack range for {pos_name}: ({z_min}, {z_max})")
        continue
    z_step = 1.5
    set_z_stack(z_min, z_max, z_step)
    add_acquisition_position(pos_name, x, y, None, None)

say("[ACTION] Configuring DAPI channel for acquisition")
add_channels('2-U-FUNA', exposure_time)

say("[ACTION] Configuring time-lapse acquisition: 24 frames, 1 hour interval")
set_time_series(24, 3600)

say("[ACTION] Starting automated multi-dimensional image acquisition")
acquisition_results = run_acquisition()
```
