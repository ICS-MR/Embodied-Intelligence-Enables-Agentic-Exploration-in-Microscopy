```python
say("[ACTION] Automatically adjusting halogen lamp brightness for brightfield imaging.")
brightness = perform_autobrightness()
if brightness < 0 or brightness > 250:
    say("[ERROR] Autobrightness returned out-of-range value. Aborting.")
    raise ValueError("Brightness out of range.")
set_brightness(brightness)
say(f"[INFO] Halogen lamp brightness set to {brightness}.")

say("[ACTION] Performing autofocus.")
optimal_z = perform_autofocus()
if optimal_z < 0 or optimal_z > 10000:
    say("[ERROR] Autofocus returned out-of-range Z position. Aborting.")
    raise ValueError("Z position out of range.")
set_z_position(optimal_z)
say(f"[INFO] Z-axis moved to autofocus position {optimal_z} μm.")

say("[ACTION] Obtaining recommended Z-stack range.")
z_max, z_min = z_stack_range()
if not (0 <= z_min < z_max <= 10000):
    say("[ERROR] Z-stack range out of hardware bounds. Aborting.")
    raise ValueError("Z-stack range out of bounds.")
say(f"[INFO] Recommended Z-stack range: {z_min} μm to {z_max} μm.")

z_mid = (z_max + z_min) / 2
say(f"[ACTION] Moving Z-axis to midpoint of recommended Z-stack range: {z_mid} μm.")
set_z_position(z_mid)
say(f"[INFO] Z-axis moved to {z_mid} μm.")
```
