cols = 3
img_h, img_w = images_np[0].shape
stitched_img = np.zeros((img_h * rows, img_w * cols), dtype=images_np[0].dtype)
for idx, img in enumerate(images_np):
    row = idx // cols
    col = idx % cols
    stitched_img[row*img_h:(row+1)*img_h, col*img_w:(col+1)*img_w] = img

say("[ACTION] Creating ImageWithMetadata for stitched image.")
stitched_meta = images_meta[0]
stitched_meta.dataset = stitched_img
stitched_meta.center_x_um = stitched_meta.center_x_um + (img_w * (cols-1) * stitched_meta.pixel_size_x_um) / 2
stitched_meta.center_y_um = stitched_meta.center_y_um + (img_h * (rows-1) * stitched_meta.pixel_size_y_um) / 2

say("[ACTION] Saving stitched DAPI image.")
save_image(stitched_meta, "stitched_dapi.ome.tif", "Stitched DAPI channel image from 3x3 grid.")
