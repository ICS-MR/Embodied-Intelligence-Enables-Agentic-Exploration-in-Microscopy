say("[ACTION] Loading Brightfield channel image.")
brightfield_meta = load_image('Brightfield_20x.ome.tif')

fitc_meta = load_image('FITC_20x.ome.tif')
say("[ACTION] Loading DAPI channel image.")
dapi_meta = load_image('DAPI_20x.ome.tif')
say("[ACTION] Loading TRITC channel image.")
tritc_meta = load_image('TRITC_20x.ome.tif')

say("[ACTION] Performing extended depth of field (Z-projection) on Brightfield channel.")
brightfield_proj = z_projection(brightfield_meta, method='Max')
say("[ACTION] Performing extended depth of field (Z-projection) on FITC channel.")
fitc_proj = z_projection(fitc_meta, method='Max')
say("[ACTION] Performing extended depth of field (Z-projection) on DAPI channel.")
dapi_proj = z_projection(dapi_meta, method='Max')
say("[ACTION] Performing extended depth of field (Z-projection) on TRITC channel.")
tritc_proj = z_projection(tritc_meta, method='Max')

save_image(brightfield_proj, 'Brightfield_20x_infocus.ome.tif', 'In-focus Brightfield image after Z-projection')

merged_meta = merge_channels(
    [brightfield_proj, fitc_proj, dapi_proj, tritc_proj],
    colors=["Brightfield", "Green", "Blue", "Red"],
    outpath="Composite_20x_RGB.ome.tif"
