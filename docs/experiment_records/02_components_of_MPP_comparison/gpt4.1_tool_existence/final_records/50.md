fiji_initialize()
say("[ACTION] Loading the OME-TIFF image file.")
image_meta = load_image('brightfield_20x_3x3mm.ome.tif')
say("[ACTION] Converting image to numpy array for processing.")
img_np = convert_to_numpy(image_meta)
if img_np.ndim == 3:
    img_np = img_np[0]
height, width = img_np.shape
pixel_size_um = 0.32
scalebar_length_um = 100
scalebar_length_px = int(scalebar_length_um / pixel_size_um)
scalebar_height_px = max(4, int(0.01 * height))
margin_px = int(0.03 * min(height, width))
scalebar_color = (255, 255, 255)
if img_np.dtype != np.uint8:
    img_disp = cv.normalize(img_np, None, 0, 255, cv.NORM_MINMAX).astype(np.uint8)
else:
    img_disp = img_np.copy()
img_rgb = cv.cvtColor(img_disp, cv.COLOR_GRAY2RGB)
start_x = margin_px
start_y = height - margin_px - scalebar_height_px
end_x = start_x + scalebar_length_px
end_y = start_y + scalebar_height_px
cv.rectangle(img_rgb, (start_x, start_y), (end_x, end_y), scalebar_color, -1)
label_text = "100 μm"
font = cv.FONT_HERSHEY_SIMPLEX
font_scale = 0.7
font_thickness = 2
(text_width, text_height), baseline = cv.getTextSize(label_text, font, font_scale, font_thickness)
text_x = start_x
text_y = start_y - 8
if text_y - text_height < 0:
    text_y = start_y + scalebar_height_px + text_height + 4
cv.putText(img_rgb, label_text, (text_x, text_y), font, font_scale, scalebar_color, font_thickness, cv.LINE_AA)
say("[ACTION] Displaying the processed image with scale bar and label for 10 seconds.")
plt.imshow(img_rgb)
plt.axis('off')
plt.show(block=False)
plt.pause(10)
plt.close()
fiji_shutdown()
