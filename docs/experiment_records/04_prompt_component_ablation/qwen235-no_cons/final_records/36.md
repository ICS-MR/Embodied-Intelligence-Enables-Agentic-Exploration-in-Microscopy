brightfield_file = "brightfield_image.ome.tif"
fitc_file = "FITC_fluorescence.ome.tif"
tritc_file = "TRITC_fluorescence.ome.tif"

say("[ACTION] Loading brightfield image: " + brightfield_file)
brightfield_image = load_image(brightfield_file)
say("[ACTION] Loading FITC fluorescence image: " + fitc_file)
fitc_image = load_image(fitc_file)
say("[ACTION] Loading TRITC fluorescence image: " + tritc_file)
tritc_image = load_image(tritc_file)

say("[ACTION] Merging brightfield, FITC, and TRITC images into RGB overlay")
merged_image = merge_channels(
    [brightfield_image, fitc_image, tritc_image],
    colors=["Brightfield", "Green", "Red"],
    outpath="merged_rgb_output.ome.tif"
)

say("[ACTION] Saving merged RGB image")
save_image(merged_image, "merged_rgb_output.ome.tif", "RGB overlay of brightfield, FITC, and TRITC channels")

fiji_shutdown()
