```python
fiji_initialize()
input_file = "3x3_mm_region.ome.tif"
say("[ACTION] Loading brightfield image: " + input_file)
image = load_image(input_file)
say("[ACTION] Applying basic processing: denoising and contrast adjustment")
denoised_image = denoise(image, method="Gaussian", radius=1.0)
processed_image = adjust_contrast(denoised_image, saturated=0.2)
scale_bar_length_px = 100  # Example scale bar length in pixels
scale_bar_um = scale_bar_length_px * processed_image.pixel_size_x_um
say(f"[ACTION] Adding scale bar of {scale_bar_um:.2f} μm (={scale_bar_length_px} pixels) for 20× magnification")
# Convert to numpy array for further processing with OpenCV/matplotlib
image_array = convert_to_numpy(processed_image)
# Draw scale bar
cv.rectangle(image_array, (10, image_array.shape[0] - 30), (10 + scale_bar_length_px, image_array.shape[0] - 10), 255, -1)
cv.putText(image_array, f'{scale_bar_um:.2f} μm', (10, image_array.shape[0] - 40),
          cv.FONT_HERSHEY_SIMPLEX, 0.5, 255, 1, cv.LINE_AA)
# Add text label
cv.putText(image_array, '20x Magnification', (10, image_array.shape[0] - 10),
          cv.FONT_HERSHEY_SIMPLEX, 0.5, 255, 1, cv.LINE_AA)
say("[ACTION] Displaying processed image with scale bar and label using matplotlib")
plt.figure()
plt.imshow(image_array, cmap='gray')
plt.axis('off')
plt.title('Processed Brightfield Image with Scale Bar and Label')
plt.show(block=False)
plt.pause(10)
plt.close()
fiji_shutdown()
```
