# Load the global view image
global_image = load_image("global_view.ome.tif")
say("[ACTION] Loaded global view image")

# Detect positions of 2D cells in the image
target_positions = analysis_platform_find_target_positions(global_image, "2Dcell", "Detected 2D cell positions")
say("[ACTION] Detected 2D cell positions")

# Analyze to find the largest 2D cell area
largest_area = 0
largest_position = None

for idx, (center_x, center_y, width, height) in enumerate(target_positions):
    area = width * height
    if area > largest_area:
        largest_area = area
        largest_position = (center_x, center_y, width, height)

if largest_position:
    say(f"[INFO] Largest 2D cell area: {largest_area} pixels² at position {largest_position}")
else:
    say("[ERROR] No 2D cells detected")

fiji_shutdown()
