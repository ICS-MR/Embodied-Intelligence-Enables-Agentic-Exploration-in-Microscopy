fiji_initialize()
input_file = "brightfield_image.ome.tif"
say("[ACTION] Loading brightfield image of 2D section: " + input_file)
image = load_image(input_file)

say("[ACTION] Applying Gaussian denoising to the image")
denoised_image = denoise(image, method="Gaussian", radius=1.0)

magnification = 20  # Extracted from image description
say("[ACTION] Performing Richardson-Lucy deconvolution with " + str(15) + " iterations")
deconvolved_image = richardson_lucy(denoised_image, magnification, iterations=15, out_filename="deconvolved_brightfield", out_dir=".")

# Convert images to numpy arrays for display
original_np = convert_to_numpy(image)
processed_np = convert_to_numpy(deconvolved_image)

# Display original and processed images side-by-side
plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.title("Original Image")
plt.imshow(original_np, cmap='gray')

plt.subplot(1, 2, 2)
plt.title("Processed Image (Denoised + Deconvolved)")
plt.imshow(processed_np, cmap='gray')

plt.show(block=False)
say("[ACTION] Displaying original and processed images side-by-side for 10 seconds")
plt.pause(10)
plt.close()

output_file = "processed_brightfield_image.ome.tif"
say("[ACTION] Saving processed image to: " + output_file)
save_image(deconvolved_image, output_file, "Processed with Gaussian denoising and RL deconvolution")

fiji_shutdown()
