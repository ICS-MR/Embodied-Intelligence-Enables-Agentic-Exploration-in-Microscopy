say("[INFO] Starting to load target bounding boxes of detected 2D cells from JSON file")

say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of detected 2D cells")

say("[INFO] Sorting 2D cell positions by area in descending order")
# Assuming bounding boxes are in format (center_x, center_y, width, height)
# Calculate area as width * height
sorted_cells = sorted(target_bounding_boxes, key=lambda x: x[2]*x[3], reverse=True)
say(f"[INFO] Sorted {len(sorted_cells)} cells by area in descending order")

say("[INFO] Moving to the first position in the sorted list")
first_cell = sorted_cells[0]
first_x, first_y = first_cell[0], first_cell[1]
