    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")

current_channel = get_channel()

say("[INFO] Loading target positions of detected organoids from JSON file")
target_filename = "organoid_locations_list.json"
target_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} organoid positions")
if len(target_bounding_boxes) > 0:
    first_box = target_bounding_boxes[0]
    first_x = first_box[0]
    first_y = first_box[1]
    say(f"[ACTION] Moving to first organoid position: X={first_x} μm, Y={first_y} μm")
    set_x_y_position(first_x, first_y)
    say("[INFO] Performing halogen lamp brightness adjustment")
    optimal_brightness = perform_autobrightness()
    set_brightness(optimal_brightness)
    say(f"[INFO] Auto-brightness adjustment complete, halogen lamp brightness set to {optimal_brightness}")
    say("[INFO] Performing autofocus on current organoid field of view")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Auto-focus complete, Z position set to {optimal_z} μm")
    say("[INFO] Determining Z-stack range for 3D organoid imaging")
    z_max, z_min = z_stack_range()
    z_step = 3.0
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)

    current_exposure = get_exposure()

    add_channels(channel=target_channel, exposure=current_exposure)
    say(f"[INFO] Acquisition parameters configured for {len(target_bounding_boxes)} organoids")
    say("[INFO] Starting automated image acquisition for all organoid positions")
    run_acquisition()
    say("[INFO] Automated image acquisition completed successfully")
else:
    say("[ERROR] No organoid positions were loaded from the file")
