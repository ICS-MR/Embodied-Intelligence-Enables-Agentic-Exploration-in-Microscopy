```python
say("[INFO] Starting imaging and cell segmentation workflow for 2D section")

say("[ACTION] Optimizing imaging parameters for current field of view")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Brightness optimized to {optimal_brightness}")

optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Focus optimized to Z position {optimal_z} μm")

say("[ACTION] Configuring image acquisition parameters")
current_x, current_y = get_x_y_position()
current_exposure = get_exposure()
current_channel = get_channel()

add_acquisition_position(name="2D_section", x=current_x, y=current_y, width=None, height=None)
add_channels(channel=current_channel, exposure=current_exposure)

say("[ACTION] Acquiring image of 2D section")
run_acquisition()
say("[INFO] Image acquisition completed")

say("[ACTION] Initializing Cellpose model for cell segmentation")
cellpose_initialize(gpu=True, model_type="cyto3")

say("[ACTION] Reading acquired image for segmentation")
image_filename = "2D_section.ome.tif"
image = cellpose_read(image_filename)
say(f"[INFO] Image loaded with shape: {image.shape}")

say("[ACTION] Preprocessing image to 2D format for segmentation")
if len(image.shape) == 5:
    image_2d = image[0, 0, 0, :, :]
elif len(image.shape) == 3:
    image_2d = image[0, :, :]
else:
    image_2d = image

say("[ACTION] Performing cell segmentation")
masks = segment(
    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say(f"[INFO] Segmentation completed. Detected {np.max(masks)} cells")

say("[ACTION] Saving segmentation masks")
mask_filename = "2D_section_segmentation_masks.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from 2D section image using Cellpose cyto3 model"
)
say(f"[INFO] Segmentation masks saved as: {mask_filename}")

say("[INFO] Cell segmentation workflow completed successfully")
```
