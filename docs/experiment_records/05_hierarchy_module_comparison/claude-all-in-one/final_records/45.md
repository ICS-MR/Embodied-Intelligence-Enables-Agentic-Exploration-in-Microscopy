say("[INFO] Starting 2D section imaging workflow for cell detection and area-based acquisition")

say("[ACTION] Adjusting imaging parameters for 2D section")
current_channel = get_channel()
if current_channel != '1-NONE':
    set_channel('1-NONE')
    say("[INFO] Filter set to brightfield mode for 2D section imaging")

current_exposure = get_exposure()
if current_exposure != 10.0:
    set_exposure(10.0)
    say("[INFO] Exposure time adjusted to 10ms for brightfield imaging")

say("[ACTION] Performing automatic brightness adjustment")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Brightness automatically adjusted to optimal value: {optimal_brightness}")

say("[ACTION] Performing autofocus on 2D section")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, Z position set to: {optimal_z} μm")

say("[ACTION] Capturing initial image for cell detection")
current_x, current_y = get_x_y_position()
add_acquisition_position(name="initial_2d_section", x=current_x, y=current_y, width=None, height=None)
add_channels(channel='1-NONE', exposure=10.0)
run_acquisition()
say("[INFO] Initial 2D section image captured")

say("[ACTION] Initializing Cellpose for 2D cell segmentation")
cellpose_initialize(gpu=False, model_type="cyto3")

say("[ACTION] Reading captured image for cell detection")
image_files = ["initial_2d_section.ome.tif"]
image = cellpose_read(image_files[0])
say(f"[INFO] Image loaded with shape: {image.shape}")

if len(image.shape) == 5:
    image_2d = image[0, 0, 0, :, :]
elif len(image.shape) == 3:
    image_2d = image[0, :, :]
else:
    image_2d = image
say("[ACTION] Performing 2D cell segmentation")
masks = segment(
    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=False
)
say(f"[INFO] Cell segmentation completed. Detected {np.max(masks)} cells")

say("[ACTION] Saving target locations sorted by cell area")
target_locations_file = save_target_locations(
    masks=masks,
    source_image_path=image_files[0],
    filename="2d_cells_by_area.json",
    description="2D cell positions sorted by area (descending order)",
    min_area_px=15,
    top_k=None
)
say(f"[INFO] Target locations saved to: {target_locations_file}")

say("[ACTION] Loading target positions for sequential acquisition")
target_positions = load_target_locations("2d_cells_by_area.json")
say(f"[INFO] Loaded {len(target_positions)} target positions sorted by cell area")

say("[ACTION] Acquiring images at each target position in descending order of cell area")
for i, (center_x, center_y, width, height) in enumerate(target_positions):
    say(f"[ACTION] Moving to target cell {i+1} at position ({center_x}, {center_y}) μm")
    set_x_y_position(center_x, center_y)

    say(f"[ACTION] Performing autofocus for target cell {i+1}")
    optimal_z_target = perform_autofocus()
    set_z_position(optimal_z_target)

    say(f"[ACTION] Capturing image of target cell {i+1}")
    add_acquisition_position(name=f"cell_{i+1}_area_rank", x=center_x, y=center_y, width=width, height=height)
    add_channels(channel='1-NONE', exposure=10.0)
    run_acquisition()
    say(f"[INFO] Image captured for target cell {i+1}")

say("[INFO] 2D section imaging workflow completed successfully")
