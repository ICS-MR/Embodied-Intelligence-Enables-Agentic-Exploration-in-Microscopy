```python
say("[INFO] Starting organoid imaging workflow in 1mm circular gel droplet")

say("[ACTION] Setting up low-magnification objective for organoid position detection")
target_objective_low = '1-UPLFLN4XPH'  # 4x objective for overview
current_objective = get_objective()
if current_objective != target_objective_low:
    set_objective(target_objective_low)
    say(f"[INFO] Objective switched to 4x (label: {target_objective_low})")
else:
    say(f"[INFO] Objective is already 4x (label: {target_objective_low}), no change needed")

say("[ACTION] Configuring brightfield mode for organoid detection")
target_channel = '1-NONE'
current_channel = get_channel()
if current_channel != target_channel:
    set_channel(target_channel)
    say(f"[INFO] Filter set to brightfield mode (channel: {target_channel})")
else:
    say(f"[INFO] Filter is already in brightfield mode (channel: {target_channel}), no change needed")

say("[ACTION] Setting appropriate exposure for brightfield imaging")
set_exposure(10.0)
say("[INFO] Exposure time set to 10ms for brightfield imaging")

say("[ACTION] Performing automatic brightness adjustment")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Brightness automatically adjusted to {optimal_brightness}")

say("[ACTION] Performing autofocus for clear organoid detection")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, Z position: {optimal_z} μm")

say("[ACTION] Capturing overview image of 1mm gel droplet for organoid detection")
current_x, current_y = get_x_y_position()
add_acquisition_position(name="organoid_overview_4x", x=current_x, y=current_y, width=1000, height=1000)
add_channels(channel=target_channel, exposure=10.0)
run_acquisition()
say("[INFO] Overview image captured successfully")

say("[ACTION] Detecting organoid positions in the captured overview image")
fiji_initialize()
overview_image = load_image("organoid_overview_4x.ome.tif")
organoid_positions = analysis_platform_find_target_positions(overview_image, "organoid", "Organoid positions detected in 1mm gel droplet at 4x magnification")
fiji_shutdown()
say(f"[INFO] Detected {len(organoid_positions)} organoid positions")

say("[ACTION] Switching to 20x objective for detailed organoid imaging")
target_objective_high = '3-LUCPLFLN20XRC'
set_objective(target_objective_high)
say(f"[INFO] Objective switched to 20x (label: {target_objective_high})")

say("[ACTION] Adjusting exposure for 20x magnification")
set_exposure(10.0)
say("[INFO] Exposure time maintained at 10ms for 20x brightfield imaging")

say("[ACTION] Performing brightness adjustment for 20x objective")
optimal_brightness_20x = perform_autobrightness()
set_brightness(optimal_brightness_20x)
say(f"[INFO] Brightness adjusted to {optimal_brightness_20x} for 20x objective")

say("[ACTION] Configuring Z-stack parameters for 3D organoid imaging")
z_step = 1.5  # Appropriate step size for 20x objective
z_range = 30  # Typical organoid thickness range
current_z = get_z_position()
z_start = current_z - z_range/2
z_end = current_z + z_range/2
set_z_stack(z_start=z_start, z_end=z_end, z_step=z_step)
say(f"[INFO] Z-stack configured: {z_start} to {z_end} μm with {z_step} μm steps")

say("[ACTION] Sequential imaging of detected organoids at 20x magnification")
for i, (center_x_px, center_y_px, width_px, height_px) in enumerate(organoid_positions):
    # Convert pixel coordinates to stage coordinates (assuming 4x pixel size ~1.6 μm/px)
    pixel_size_4x = 1.6
    organoid_x = current_x + (center_x_px - 512) * pixel_size_4x  # Assuming 1024x1024 image center
    organoid_y = current_y + (center_y_px - 512) * pixel_size_4x

    say(f"[ACTION] Moving to organoid {i+1} at position ({organoid_x:.1f}, {organoid_y:.1f}) μm")
    set_x_y_position(organoid_x, organoid_y)

    say(f"[ACTION] Performing autofocus for organoid {i+1}")
    optimal_z_organoid = perform_autofocus()
    set_z_position(optimal_z_organoid)

    # Update Z-stack center based on focused position
    z_start_organoid = optimal_z_organoid - z_range/2
    z_end_organoid = optimal_z_organoid + z_range/2
    set_z_stack(z_start=z_start_organoid, z_end=z_end_organoid, z_step=z_step)

    say(f"[ACTION] Acquiring Z-stack for organoid {i+1}")
    add_acquisition_position(name=f"organoid_{i+1}_20x", x=organoid_x, y=organoid_y, width=None, height=None)
    add_channels(channel=target_channel, exposure=10.0)
    run_acquisition()
    say(f"[INFO] Organoid {i+1} Z-stack acquisition completed")

say("[INFO] Organoid imaging workflow completed successfully")
say(f"[INFO] Total organoids imaged: {len(organoid_positions)}")
```
