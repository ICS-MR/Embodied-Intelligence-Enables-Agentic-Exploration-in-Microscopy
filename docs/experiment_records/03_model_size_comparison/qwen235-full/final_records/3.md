say("[ACTION] Initializing Cellpose2 model with 'cyto' model for cytoplasm segmentation and GPU acceleration")
cellpose_initialize(model_type="cyto", gpu=True)

say("[ACTION] Reading 20x brightfield image from microscope output")
image_path = 'current_field.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing brightfield image to 2D format")
# Extract HxW from TCZYX (T=0, C=0, Z=0)
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Executing Cellpose2 cytoplasm segmentation")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for brightfield
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Detected {} cells in brightfield image".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "cytoplasm_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from 20x brightfield image using Cellpose2 'cyto' model"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[INFO] Workflow completed successfully. Model resources automatically released.")
