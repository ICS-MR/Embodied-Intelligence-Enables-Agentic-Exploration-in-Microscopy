cellpose_initialize(model_type="cpsam")

image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preparing 2D image from DAPI fluorescence data")
# Extract 2D image from TCZYX dimensions (T=0, C=0, Z=0)
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Performing nucleus segmentation")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for nucleus segmentation
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} nuclei".format(np.max(masks)))

say("[ACTION] Saving nucleus segmentation masks")
mask_filename = "nucleus_segmentation_3.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Nucleus segmentation masks from DAPI fluorescence image of 3x3 mm area"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing nucleus size distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False
)

analysis_filename = "nucleus_analysis_3.csv"
