say("[ACTION] Initializing Cellpose model for labeled cell counting with GPU acceleration")
cellpose_initialize(model_type="cpsam", gpu=True)

say("[ACTION] Reading stitched DAPI fluorescence image from context")
image_path = 'dapi_fluorescence.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Extracting DAPI channel (C=0) from TCZYX data")
# TCZYX -> select T=0, Z=0, C=0 (DAPI channel), then HxW
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Running labeled cell segmentation on DAPI channel")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for nuclei
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} labeled cells".format(np.max(masks)))

say("[ACTION] Saving labeled cell segmentation masks")
mask_filename = "labeled_cell_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Labeled cell segmentation masks from DAPI fluorescence image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing total number of labeled cells")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=1,  # Only count total cells
    plot=False
)

total_cells = len(analysis_df)
say("[INFO] Total number of labeled cells detected: {}".format(total_cells))

say("[ACTION] Saving labeled cell count analysis results")
analysis_filename = "labeled_cell_count_analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
