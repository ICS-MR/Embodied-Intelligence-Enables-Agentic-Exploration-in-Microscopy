cellpose_initialize(model_type="cyto", gpu=True)

image_path = 'brightfield_current.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format")
# Extract T=0, C=0, Z=0 (assuming single channel brightfield image) -> HxW
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Running cell segmentation on brightfield image")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Brightfield only, no nucleus channel
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Saving cell segmentation masks")
mask_filename = "cell_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from brightfield_current image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing cell area distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False

analysis_filename = "cell_analysis_1.csv"
