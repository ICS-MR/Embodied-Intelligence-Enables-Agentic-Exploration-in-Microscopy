```python
fiji_initialize()
input_file = "brightfield_3x3mm.ome.tif"
say("[ACTION] Loading brightfield image of the 3 × 3 mm region: " + input_file)
image = load_image(input_file)

# Convert to numpy array for processing
image_array = convert_to_numpy(image)

# Add scale bar and text label in the lower-left corner
pixel_size_um = image.pixel_size_x_um  # Assuming square pixels
scale_bar_length_px = int(50 / pixel_size_um)  # 50 micrometers in pixels

# Draw scale bar (white line)
cv.rectangle(image_array, (10, image_array.shape[0] - 20), (10 + scale_bar_length_px, image_array.shape[0] - 10), (255, 255, 255), 1)

# Add text label
font = cv.FONT_HERSHEY_SIMPLEX
font_scale = 0.5
font_thickness = 1
text = f"{int(pixel_size_um)} µm/pixel"
text_size, _ = cv.getTextSize(text, font, font_scale, font_thickness)
text_origin = (10, image_array.shape[0] - 5)
cv.putText(image_array, text, text_origin, font, font_scale, (255, 255, 255), font_thickness)

# Display the processed image using plt
say("[ACTION] Displaying processed image with scale bar and text label")
plt.imshow(image_array, cmap='gray')
plt.title("Brightfield Image with Scale Bar")
plt.axis('off')  # Hide axes
plt.show(block=False)
plt.pause(10)  # Display for 10 seconds
plt.close()

fiji_shutdown()
```
