masks = segment(
    image=image_2d,
    channels=[0, 0],  # Assuming single channel for cell segmentation
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "cell_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from current_state_image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing cell area distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False

analysis_filename = "cell_analysis_1.csv"
