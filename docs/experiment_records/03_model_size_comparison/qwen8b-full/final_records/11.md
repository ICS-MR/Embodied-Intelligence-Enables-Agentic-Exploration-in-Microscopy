import json

fiji_initialize()

# Load the JSON file to verify its existence and extract target positions
json_file = "2Dcell_locations_list.json"
say("[ACTION] Importing detected-region JSON file: " + json_file)
try:
    with open(json_file, 'r') as file:
        target_positions_data = json.load(file)
except FileNotFoundError:
    say("[ERROR] The file " + json_file + " was not found.")
    fiji_shutdown()

# Check if the JSON data contains at least one detected region
if not target_positions_data or len(target_positions_data) == 0:
    say("[ERROR] No target positions found in the JSON file.")
    fiji_shutdown()

# Load the target position bounding boxes of the first detected region
first_detected_region = target_positions_data[0]
bounding_box = first_detected_region.get("bounding_box", None)

if bounding_box is None:
    say("[ERROR] The first detected region does not contain a bounding box.")
    fiji_shutdown()

# Output the bounding box coordinates
center_x_px, center_y_px, width_px, height_px = bounding_box
say("[INFO] Target position bounding box of the first detected region:")
say("[INFO] Center X (px): " + str(center_x_px))
say("[INFO] Center Y (px): " + str(center_y_px))
say("[INFO] Width (px): " + str(width_px))
say("[INFO] Height (px): " + str(height_px))

fiji_shutdown()
