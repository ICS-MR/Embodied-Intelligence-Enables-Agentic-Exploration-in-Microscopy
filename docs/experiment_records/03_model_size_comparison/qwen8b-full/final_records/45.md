#### Microscope Operation Platform Code 4
```python
say("[INFO] Starting to load 2D cell bounding boxes from JSON file")
target_filename = "2Dcell_locations_list.json"
say(f"[ACTION] Loading target locations from file {target_filename}")
target_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of 2D cells")

# Sort the bounding boxes in descending order of area
target_bounding_boxes.sort(key=lambda x: x[2] * x[3], reverse=True)

say("[INFO] Starting iterative operation for 2D cell imaging")
for i, (center_x, center_y, width, height) in enumerate(target_bounding_boxes):
    say(f"[ACTION] Processing cell {i + 1} with bounding box: ({center_x}, {center_y}, {width}, {height})")
    
    # Move to the center position of the bounding box
    say(f"[ACTION] Moving to cell position: X={center_x} μm, Y={center_y} μm")
    set_x_y_position(center_x, center_y)
    
    # Adjust brightness for brightfield imaging
    say("[INFO] Performing auto-brightness adjustment for brightfield")
    optimal_brightness = perform_autobrightness()
    set_brightness(optimal_brightness)
    say(f"[INFO] Auto-brightness adjustment complete, halogen lamp brightness set to {optimal_brightness}")
    
    # Perform autofocus on the current field of view
    say("[INFO] Performing autofocus")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Auto-focus completed, Z-axis position: {optimal_z} μm")
    
    # Configure acquisition parameters for brightfield imaging
    say("[INFO] Configuring brightfield image acquisition parameters")
    current_x, current_y = get_x_y_position()
    say(f"[INFO] Current XY position: ({current_x}, {current_y}) μm")
    add_acquisition_position(name=f"cell_{i + 1}", x=current_x, y=current_y, width=width, height=height)
    current_exposure = get_exposure()
    add_channels(channel='1-NONE', exposure=current_exposure)
    say(f"[INFO] Brightfield acquisition parameters configured: channel '1-NONE', exposure {current_exposure} ms")
    
    # Perform automated image acquisition
    say("[INFO] Initiating automated image acquisition")
    run_acquisition()
    say("[INFO] Automated image acquisition task completed for cell {i + 1}")
```
