fiji_initialize()
input_file = "brightfield_3x3mm.ome.tif"

say("[ACTION] Adding scale bar and text label to the image")
# Convert image to numpy array for processing
image_array = convert_to_numpy(image)
# Add scale bar and text label
height, width = image_array.shape
scale_bar_length_px = 100  # Example scale bar length in pixels
scale_bar_position = (width - scale_bar_length_px - 10, height - 10)  # Position at bottom-right corner
text_position = (10, height - 10)  # Position at bottom-left corner

# Draw scale bar
cv2.line(image_array, scale_bar_position, (scale_bar_position[0] + scale_bar_length_px, scale_bar_position[1]), (255, 255, 255), 2)
# Add text label
cv2.putText(image_array, "Scale bar", (scale_bar_position[0] + scale_bar_length_px + 5, scale_bar_position[1] + 15), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)

# Add text label
cv2.putText(image_array, "Brightfield 3x3mm", text_position, cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)

# Save processed image
output_file = "processed_brightfield_3x3mm.ome.tif"
say("[ACTION] Saving processed image with scale bar and text label to: " + output_file)
save_image(image, output_file, "Processed image with scale bar and text label")

# Display the processed image using plt
import matplotlib.pyplot as plt
