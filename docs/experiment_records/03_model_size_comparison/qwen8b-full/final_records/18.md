say("[INFO] Starting parameter setup: 10x objective and brightfield mode")
target_objective = '2-SOB'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 10x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 10x (label: {target_objective}), no change needed")
target_channel = '1-NONE'
current_channel = get_channel()
if current_channel != target_channel:
    set_channel(target_channel)
    say(f"[INFO] Filter set to brightfield mode (channel: {target_channel})")
else:
    say(f"[INFO] Filter is already in brightfield mode (channel: {target_channel}), no change needed")
say("[INFO] Automatically configuring camera exposure time for brightfield imaging")
current_exposure = get_exposure()
say(f"[INFO] Current exposure time: {current_exposure} ms")
say("[INFO] Adjusting halogen lamp brightness to an appropriate nonzero level")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Halogen lamp brightness set to optimal value: {optimal_brightness}")
say("[INFO] Performing autofocus on current field of view")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Auto-focus completed, Z-axis position: {optimal_z} μm")
say("[INFO] Configuring image acquisition parameters for 4×4 mm region under brightfield illumination")
current_x, current_y = get_x_y_position()
say(f"[INFO] Current XY position: ({current_x}, {current_y}) μm")
width = 4000  # Convert 4 mm to micrometers
height = 4000  # Convert 4 mm to micrometers
say(f"[INFO] Acquisition size set to {width} μm × {height} μm to cover 4×4 mm region")
add_acquisition_position(name="brightfield_10x_4x4mm", x=current_x, y=current_y, width=width, height=height)
add_channels(channel=target_channel, exposure=current_exposure)
say("[INFO] Brightfield acquisition parameters configured")
say("[INFO] Initiating automated image acquisition for 4×4 mm region under brightfield illumination")
run_acquisition()
say("[INFO] Automated image acquisition task completed")
