say("[INFO] Starting autofocus on current field of view")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, Z position set to {optimal_z} μm")
say("[INFO] Setting objective lens to 20x")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
say("[INFO] Automatically configuring camera exposure time for TMRM fluorescence mode")
target_exposure_tmr = 300.0
set_exposure(target_exposure_tmr)
say(f"[INFO] Camera exposure time set to {target_exposure_tmr} ms for TMRM fluorescence")
set_brightness(0)
say("[INFO] Halogen lamp brightness set to 0 for fluorescence imaging")
say("[INFO] Configuring multi-channel acquisition parameters for TMRM, Hoechst, and Calcein")
current_x, current_y = get_x_y_position()
say(f"[INFO] Current XY position: ({current_x}, {current_y}) μm")
add_acquisition_position(name="multi_channel_fluorescence", x=current_x, y=current_y, width=None, height=None)
add_channels(channel='2-U-FUNA', exposure=target_exposure_tmr)
add_channels(channel='3-U-FBNA', exposure=250.0)
add_channels(channel='4-U-FGNA', exposure=200.0)
say("[INFO] Multi-channel fluorescence acquisition parameters configured")
say("[INFO] Initiating automated image acquisition for TMRM, Hoechst, and Calcein channels")
run_acquisition()
say("[INFO] Automated image acquisition for multi-channel fluorescence completed")
