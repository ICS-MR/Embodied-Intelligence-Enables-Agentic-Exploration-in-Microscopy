say("[INFO] Starting to load target bounding boxes of 2D cells from JSON file")
target_filename = "2Dcell_locations_list.json"

say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of 2D cells")
say("[INFO] Sorting bounding boxes in descending order of cell area")
sorted_bounding_boxes = sorted(target_bounding_boxes, key=lambda box: box[2]*box[3], reverse=True)

current_exposure = get_exposure()
target_channel = '1-NONE'

    position_name = f"cell_{idx+1:03d}"
    add_acquisition_position(name=position_name, x=center_x, y=center_y, width=width, height=height)
say("[INFO] Acquisition positions for all 2D cells have been added")
add_channels(channel=target_channel, exposure=current_exposure)
