say("[INFO] Starting objective lens parameter setup")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
say("[INFO] Loading organoid target positions from JSON file")
target_filename = "organoid_locations_list.json"
say(f"[ACTION] Loading organoid locations from file {target_filename}")
organoid_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(organoid_bounding_boxes)} organoid bounding boxes")
current_pixel_size = 1.62  # micrometers per pixel from metadata
image_width = 2048 * current_pixel_size
image_height = 2048 * current_pixel_size
say("[INFO] Starting iterative operations for each detected organoid")
for idx, (center_x, center_y, width, height) in enumerate(organoid_bounding_boxes, start=1):
    say(f"[ACTION] Processing organoid {idx}/{len(organoid_bounding_boxes)}")
    say(f"[INFO] Moving to organoid XY coordinates: X={center_x}, Y={center_y}")
    set_x_y_position(center_x, center_y)
    say("[INFO] Verifying organoid is centered in field of view")
    current_x, current_y = get_x_y_position()
    x_diff = center_x - current_x
    y_diff = center_y - current_y
    if abs(x_diff) > image_width/10 or abs(y_diff) > image_height/10:
        new_x = center_x
        new_y = center_y
        set_x_y_position(new_x, new_y)
        say(f"[INFO] Organoid centered: moved to new position X={new_x}, Y={new_y}")
    else:
        say("[INFO] Organoid already centered in field of view")
    say("[INFO] Performing autofocus on current organoid")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Auto-focus completed, Z position set to {optimal_z} μm")
    say("[INFO] Determining Z-stack range for 3D imaging")
    z_max, z_min = z_stack_range()
    z_step = 1.5
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)
    say(f"[INFO] Z-stack parameters set: {z_min} μm to {z_max} μm with {z_step} μm step")
    say("[INFO] Configuring fluorescence channels for DAPI, FITC, TRITC")
    current_x, current_y = get_x_y_position()
    add_acquisition_position(name=f"organoid_{idx:03d}", x=current_x, y=current_y, width=None, height=None)
    add_channels(channel='2-U-FUNA', exposure=200.0)
    add_channels(channel='3-U-FBNA', exposure=200.0)
    add_channels(channel='4-U-FGNA', exposure=200.0)
    say("[INFO] Fluorescence channels configured: DAPI (blue), FITC (green), TRITC (red)")
    say("[INFO] Initiating multi-channel fluorescence image acquisition")
    run_acquisition()
    say("[INFO] Multi-channel fluorescence image acquisition completed for this organoid")
say("[INFO] All organoid imaging operations completed successfully")
