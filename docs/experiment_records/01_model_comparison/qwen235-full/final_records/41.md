fiji_initialize()

# Import Z-stack images
brightfield_file = "brightfield_zstack.ome.tif"

say("[ACTION] Loading brightfield Z-stack image: " + brightfield_file)
brightfield_image = load_image(brightfield_file)

say("[ACTION] Loading DAPI Z-stack image: " + dapi_file)
dapi_image = load_image(dapi_file)

say("[ACTION] Loading FITC Z-stack image: " + fitc_file)
fitc_image = load_image(fitc_file)

say("[ACTION] Loading TRITC Z-stack image: " + tritc_file)
tritc_image = load_image(tritc_file)

# Extended Depth of Field processing
say("[ACTION] Performing extended depth of field processing on brightfield Z-stack")
brightfield_edof = z_projection(brightfield_image, method="max")

say("[ACTION] Performing extended depth of field processing on DAPI Z-stack")
dapi_edof = z_projection(dapi_image, method="max")

say("[ACTION] Performing extended depth of field processing on FITC Z-stack")
fitc_edof = z_projection(fitc_image, method="max")

say("[ACTION] Performing extended depth of field processing on TRITC Z-stack")
tritc_edof = z_projection(tritc_image, method="max")

# Save extended depth of field images
say("[ACTION] Saving extended depth of field brightfield image")
save_image(brightfield_edof, "edof_brightfield.tif", "Extended depth of field brightfield image")

say("[ACTION] Saving extended depth of field DAPI image")
save_image(dapi_edof, "edof_dapi.tif", "Extended depth of field DAPI image")

save_image(fitc_edof, "edof_fitc.tif", "Extended depth of field FITC image")

say("[ACTION] Saving extended depth of field TRITC image")
save_image(tritc_edof, "edof_tritc.tif", "Extended depth of field TRITC image")

# Merge fluorescence images with brightfield for overlay
say("[ACTION] Merging extended depth of field fluorescence channels with brightfield background")

# Convert to list for merge_channels function
fluorescence_channels = [dapi_edof, fitc_edof, tritc_edof]

overlay_image = merge_channels(fluorescence_channels, colors=channel_colors, outpath="fluorescence_overlay.tif")
