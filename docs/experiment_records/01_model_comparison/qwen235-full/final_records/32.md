say("[ACTION] Initializing Cellpose model for cell segmentation with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading brightfield image of the entire culture dish")

image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format")
image_2d = image[0, 0, 0, :, :]  # Extract HxW from TCZYX

say("[ACTION] Running cell segmentation inference")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Assuming grayscale image for cell identification
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Analyzing cell masks for counting")
analysis_df = analyze_masks(
    masks=masks,
    px_size=0.32,
    unit="px",
    bins=1,
    plot=False
)
cell_count = len(analysis_df[analysis_df['bin_idx'] >= 0])
say("[INFO] Total cell count: {}".format(cell_count))

say("[ACTION] Saving cell counting analysis results")
analysis_filename = "cell_counting_results.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
