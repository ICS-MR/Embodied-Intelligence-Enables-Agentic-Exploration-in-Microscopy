fiji_initialize()
brightfield_file = "brightfield_20x.ome.tif"
fluorescence_file = "fluorescence_20x.ome.tif"
say("[ACTION] Loading brightfield image: " + brightfield_file)
brightfield_image = load_image(brightfield_file)
say("[ACTION] Loading fluorescence image: " + fluorescence_file)
fluorescence_image = load_image(fluorescence_file)
say("[ACTION] Splitting multi-channel fluorescence image into individual channels")
fluorescence_channels = split_channels(fluorescence_image)
# Assuming the order of channels in fluorescence_20x.ome.tif is DAPI (Blue), FITC (Green), TRITC (Red)
# based on the description 'channel_names: [(0, 0, 255), (0, 255, 0), (255, 0, 0)]'
dapi_channel = fluorescence_channels[0]
fitc_channel = fluorescence_channels[1]
tritc_channel = fluorescence_channels[2]
merged_images = [brightfield_image, dapi_channel, fitc_channel, tritc_channel]
merged_colors = ["Brightfield", "Blue", "Green", "Red"]
output_file = "merged_brightfield_fluorescence_20x.ome.tif"
say("[ACTION] Merging brightfield image with DAPI, FITC, and TRITC fluorescence images into an RGB superimposed image: " + output_file)
merged_composite_image = merge_channels(merged_images, colors=merged_colors, outpath=output_file)
say("[ACTION] Saving merged composite image to: " + output_file)
save_image(merged_composite_image, output_file, "Merged brightfield, DAPI (Blue), FITC (Green), TRITC (Red) composite image")
fiji_shutdown()
