say("[ACTION] Initializing Cellpose model for nucleus segmentation with GPU acceleration")
cellpose_initialize(gpu=True, model_type="cpsam")

say("[ACTION] Reading DAPI-labeled cell nuclei image from context")
image_path = 'DAPI_cells.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format for segmentation")
# Assuming DAPI is the first (and likely only) channel, and we need a 2D slice (H, W)
# from the TCZYX format.
# T=0, C=0, Z=0 for a single timepoint, single channel, single Z-slice image.
image_2d = image[0, 0, 0, :, :]
say("[INFO] Extracted 2D image with shape: {}".format(image_2d.shape))

say("[ACTION] Running cell nucleus segmentation inference")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # DAPI is typically a single channel for nucleus
    diameter=None,    # Let Cellpose estimate diameter
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True      # Denoise for better segmentation
)
say("[INFO] Segmentation completed. Detected {} cell nuclei.".format(np.max(masks)))

say("[ACTION] Saving nucleus segmentation masks")
mask_filename = "nucleus_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Nucleus segmentation masks from DAPI_cells.ome.tif"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing segmented cell nucleus masks for area and quantity distribution")
# Extract pixel size from the image description
px_size = 1.62
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False # Do not plot, just return DataFrame
)
say("[INFO] Analysis completed. Generated DataFrame with {} entries.".format(len(analysis_df)))

say("[ACTION] Saving analysis results of cell nucleus area and quantity distribution")
analysis_filename = "nucleus_analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully. Model resources implicitly released.")
