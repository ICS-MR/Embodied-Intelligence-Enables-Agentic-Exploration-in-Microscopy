say("[INFO] Starting parameter setup for multi-channel, time-lapse, Z-stack acquisition of organoids")

say("[INFO] Setting filter set to FITC fluorescence mode")
target_channel_fitc = '3-U-FBNA'
current_channel = get_channel()
if current_channel != target_channel_fitc:
    set_channel(target_channel_fitc)
    say(f"[INFO] Filter set to FITC fluorescence mode (channel: {target_channel_fitc})")
else:
    say(f"[INFO] Filter is already in FITC fluorescence mode (channel: {target_channel_fitc}), no change needed")

# As there is no 'perform_autoexposure' API, setting a suitable default exposure for fluorescence.
default_fluorescence_exposure = 200.0
set_exposure(default_fluorescence_exposure)
say(f"[INFO] Camera exposure time set to a default suitable for fluorescence imaging: {default_fluorescence_exposure} ms")

set_brightness(0)
say("[INFO] Halogen lamp brightness set to 0 for fluorescence imaging")

say("[INFO] Performing automatic focusing on the current field of view containing organoids")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, optimal Z position: {optimal_z} μm")

say("[INFO] Analyzing current field of view to obtain Z-axis stack range for organoids")
z_max, z_min = z_stack_range()
# Determine Z-step based on the current objective (20x)
z_step_20x = 1.5
say(f"[INFO] Recommended Z-stack range: {z_min} μm to {z_max} μm with step {z_step_20x} μm for 20x objective")

say("[INFO] Configuring automatic image acquisition parameters")
current_x, current_y = get_x_y_position()
say(f"[INFO] Current XY position: ({current_x}, {current_y}) μm")
add_acquisition_position(name="organoid_multi_channel_time_lapse", x=current_x, y=current_y, width=None, height=None)
say("[INFO] Acquisition position parameters added for current field of view")

say("[INFO] Configuring DAPI, FITC, and TRITC fluorescence channels with default exposure parameters")
# As there is no 'perform_autoexposure' API, using a suitable default exposure for each fluorescence channel.
add_channels(channel='2-U-FUNA', exposure=default_fluorescence_exposure) # DAPI
add_channels(channel='3-U-FBNA', exposure=default_fluorescence_exposure) # FITC
add_channels(channel='4-U-FGNA', exposure=default_fluorescence_exposure) # TRITC
say(f"[INFO] Channels configured: DAPI, FITC, TRITC with default exposure {default_fluorescence_exposure} ms")

set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step_20x)
say(f"[INFO] Z-stack parameters set: range {z_min} to {z_max} μm with step {z_step_20x} μm")

total_duration_hours = 6
interval_minutes = 30
num_frames = int((total_duration_hours * 60) / interval_minutes) + 1 # +1 to include the initial frame
interval_sec = interval_minutes * 60
set_time_series(num_frames=num_frames, interval_sec=interval_sec)
say(f"[INFO] Time series configured: {num_frames} frames over {total_duration_hours} hours with {interval_minutes}-minute intervals ({interval_sec} seconds)")

say("[INFO] Initiating automated image acquisition for multi-fluorescent labels of organoids")
run_acquisition()
say("[INFO] Automated image acquisition completed successfully")
