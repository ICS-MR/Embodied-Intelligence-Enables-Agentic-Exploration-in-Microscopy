fiji_initialize()
input_file = "brightfield_3x3mm.ome.tif"
say("[ACTION] Loading 3x3mm brightfield image: " + input_file)
image_meta = load_image(input_file)
say("[ACTION] Converting image to numpy array for processing")
image_np = convert_to_numpy(image_meta)
scale_bar_length_um = 100
pixel_size_um = image_meta.pixel_size_x_um
scale_bar_length_px = int(scale_bar_length_um / pixel_size_um)
height, width = image_np.shape
bar_thickness = max(2, int(height * 0.01))
bar_color = 255
text_color = 255
font = cv.FONT_HERSHEY_SIMPLEX
font_scale = max(0.5, height / 1000)
font_thickness = max(1, int(height / 500))
bar_start_x = int(width * 0.05)
bar_end_x = bar_start_x + scale_bar_length_px
bar_y = int(height * 0.95)
cv.rectangle(image_np, (bar_start_x, bar_y - bar_thickness), (bar_end_x, bar_y), bar_color, -1)
text_label = "Scale: 100 µm"
text_size = cv.getTextSize(text_label, font, font_scale, font_thickness)[0]
text_x = bar_start_x
text_y = bar_y - bar_thickness - 10
cv.putText(image_np, text_label, (text_x, text_y), font, font_scale, text_color, font_thickness, cv.LINE_AA)
label_text = "Brightfield 3x3mm"
label_text_size = cv.getTextSize(label_text, font, font_scale, font_thickness)[0]
label_x = int(width * 0.05)
label_y = int(height * 0.05) + label_text_size[1]
cv.putText(image_np, label_text, (label_x, label_y), font, font_scale, text_color, font_thickness, cv.LINE_AA)
say("[ACTION] Displaying image with scale bar and text label for 10 seconds")
plt.imshow(image_np, cmap='gray')
plt.axis('off')
plt.title("Processed Image")
plt.pause(10)
plt.close()
fiji_shutdown()
