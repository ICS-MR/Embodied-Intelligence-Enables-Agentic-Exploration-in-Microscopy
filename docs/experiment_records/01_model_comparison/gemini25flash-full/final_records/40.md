say("[INFO] Starting parameter setup for full section scan")

if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 4x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 4x (label: {target_objective}), no change needed")

target_channel = '1-NONE'
current_channel = get_channel()
if current_channel != target_channel:
    set_channel(target_channel)
    say(f"[INFO] Filter set to Brightfield mode (channel: {target_channel})")
else:
    say(f"[INFO] Filter is already in Brightfield mode (channel: {target_channel}), no change needed")

say("[INFO] Performing auxiliary operations: adjusting light source brightness and autofocus")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Halogen lamp brightness automatically adjusted to {optimal_brightness}")

optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, optimal Z position: {optimal_z} μm")

say("[INFO] Configuring automatic image acquisition parameters for full section scan")
current_exposure = get_exposure()
say(f"[INFO] Using current camera exposure time: {current_exposure} ms")

# To cover the entire section, we assume the maximum stage movement area (0 to 500000 for X and Y)
# The acquisition position will start at (0,0) and cover the full stage dimensions.
scan_start_x = 0.0
scan_start_y = 0.0
scan_width = 500000.0
scan_height = 500000.0
say(f"[INFO] Configuring XY position to cover the entire section from ({scan_start_x}, {scan_start_y}) μm with width {scan_width} μm and height {scan_height} μm")
add_acquisition_position(name="full_section_scan", x=scan_start_x, y=scan_start_y, width=scan_width, height=scan_height)

add_channels(channel=target_channel, exposure=current_exposure)
say(f"[INFO] Brightfield channel configured with exposure {current_exposure} ms")

say("[INFO] Initiating automatic image acquisition for a full scan of the entire section")
run_acquisition()
say("[INFO] Automatic image acquisition of the entire section completed")
