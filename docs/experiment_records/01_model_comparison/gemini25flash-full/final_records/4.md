#### Microscope Operation Platform Code 1
```python
say("[INFO] Moving to the middle of the Z-axis")
current_z = get_z_position()
middle_z = 5000
set_z_position(middle_z)
say(f"[INFO] Z-axis moved from {current_z} μm to {middle_z} μm")
say("[INFO] Setting objective lens to 20x and filter to brightfield mode")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
target_channel = '1-NONE'
current_channel = get_channel()
if current_channel != target_channel:
    set_channel(target_channel)
    say(f"[INFO] Filter set to brightfield mode (channel: {target_channel})")
else:
    say(f"[INFO] Filter is already in brightfield mode (channel: {target_channel}), no change needed")
say("[INFO] Adjusting light source brightness to an appropriate level")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Halogen lamp brightness automatically adjusted to {optimal_brightness}")
say("[INFO] Performing auto-focus")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Auto-focus completed, Z position set to optimal value: {optimal_z} μm")
say("[INFO] Configuring image acquisition parameters for brightfield")
current_exposure = get_exposure()
current_x, current_y = get_x_y_position()
add_acquisition_position(name="brightfield_current_fov", x=current_x, y=current_y, width=None, height=None)
add_channels(channel='1-NONE', exposure=current_exposure)
say("[INFO] Brightfield acquisition parameters configured: channel '1-NONE', exposure set to current value")
say("[INFO] Performing automatic image acquisition")
run_acquisition()
say("[INFO] Automatic image acquisition completed")
```

fiji_initialize()

denoised_image = denoise(image, method="gaussian", radius=1.0)
output_file = "denoised_brightfield_current_fov.ome.tif"
say("[ACTION] Saving denoised image to: " + output_file)
save_image(denoised_image, output_file, "Brightfield image denoised with Gaussian filter")
fiji_shutdown()
