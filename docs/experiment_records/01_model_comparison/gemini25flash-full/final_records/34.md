    say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} organoid bounding boxes")
    first_organoid_x, first_organoid_y, _, _ = target_bounding_boxes[0]

    say("[INFO] Initiating automated image acquisition for all detected organoid regions")
    run_acquisition()
    say("[INFO] Automated image acquisition completed successfully")
