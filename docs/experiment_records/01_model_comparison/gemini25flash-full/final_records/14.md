image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format (assuming single channel DAPI image)")
# Assuming the DAPI image is already a single channel, potentially (T, C, Z, H, W) or (H, W)
# We'll extract the HxW plane, assuming T=0, C=0, Z=0 if higher dimensions exist.
if image.ndim == 5:
    image_2d = image[0, 0, 0, :, :]
elif image.ndim == 2:
    image_2d = image
else:
    say(f"[ERROR] Unexpected image dimensions: {image.shape}. Attempting to extract 2D plane.")
    # Try to extract the last two dimensions as HxW
    image_2d = image[tuple(0 for _ in range(image.ndim - 2))]

say("[ACTION] Running nucleus segmentation on DAPI/blue channel")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for nucleus
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} nuclei".format(np.max(masks)))
