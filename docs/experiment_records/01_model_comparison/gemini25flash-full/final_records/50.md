fiji_initialize()
input_file = "current_fov.ome.tif"
say("[ACTION] Loading image of the 2D section: " + input_file)
image_meta = load_image(input_file)
say("[ACTION] Detecting 2D cell regions in the imported image")
bounding_boxes = analysis_platform_find_target_positions(image_meta, "2Dcell", "Detected 2D cell regions")
say("[ACTION] Converting image to numpy array for annotation")
image_np = convert_to_numpy(image_meta)
say("[ACTION] Annotating detected areas with bounding boxes")
annotated_image_np = image_np.copy()
for bbox in bounding_boxes:
    center_x_px, center_y_px, width_px, height_px = bbox
    x1 = int(center_x_px - width_px / 2)
    y1 = int(center_y_px - height_px / 2)
    x2 = int(center_x_px + width_px / 2)
    y2 = int(center_y_px + height_px / 2)
    cv.rectangle(annotated_image_np, (x1, y1), (x2, y2), (255, 0, 0), 2)
say("[ACTION] Displaying annotated image for 5 seconds")
plt.imshow(annotated_image_np, cmap='gray')
plt.title("Detected 2D Cell Regions")
plt.pause(5)
plt.close()
fiji_shutdown()
