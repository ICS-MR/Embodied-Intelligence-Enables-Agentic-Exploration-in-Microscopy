say("[INFO] Starting auxiliary operations for optimal imaging")
say("[INFO] Automatically adjusting halogen lamp brightness")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)

say("[INFO] Performing autofocus on the current field of view")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus completed, optimal Z position: {optimal_z} μm")

say("[INFO] Analyzing current field of view to obtain Z-axis stack range for 3D cells")
z_max, z_min = z_stack_range()
current_objective = get_objective()
z_step = 0.0
if current_objective == '1-UPLFLN4XPH':
    z_step = 7.5
elif current_objective == '2-SOB':
    z_step = 3.0
elif current_objective == '3-LUCPLFLN20XRC':
    z_step = 1.5
elif current_objective == '4-LUCPLFLN40X':
    z_step = 0.75
elif current_objective == '5-LUCPLFLN60X':
    z_step = 0.5
elif current_objective == '6-UPLSAPO30XS':
    z_step = 1.0
else:
    say("[ERROR] Unknown objective, cannot determine Z-step. Using default 1.5 μm.")
    z_step = 1.5
say(f"[INFO] Recommended Z-stack range: {z_min} μm to {z_max} μm with step {z_step} μm for objective {current_objective}")

say("[INFO] Configuring image automatic acquisition parameters")
target_channel = '1-NONE'
current_exposure = get_exposure()
say(f"[INFO] Configuring brightfield channel '{target_channel}' with current exposure {current_exposure} ms")
add_channels(channel=target_channel, exposure=current_exposure)

current_x, current_y = get_x_y_position()
say(f"[INFO] Configuring XY position to current coordinates ({current_x}, {current_y}) μm, matching current field of view size")
add_acquisition_position(name="live_cell_migration", x=current_x, y=current_y, width=None, height=None)

say(f"[INFO] Configuring Z-axis stack parameters: z_start={z_min} μm, z_end={z_max} μm, z_step={z_step} μm")
set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)

total_duration_hours = 12
interval_minutes = 5
num_frames = int((total_duration_hours * 60) / interval_minutes)
interval_sec = interval_minutes * 60
say(f"[INFO] Configuring time parameters for a total duration of {total_duration_hours} hours ({num_frames} frames) with an acquisition interval of {interval_minutes} minutes ({interval_sec} seconds)")
set_time_series(num_frames=num_frames, interval_sec=interval_sec)

say("[INFO] Performing automatic image acquisition to record the migration process of living cells")
run_acquisition()
say("[INFO] Automated image acquisition task completed")
