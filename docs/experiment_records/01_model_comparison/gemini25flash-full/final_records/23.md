target_objective = '3-LUCPLFLN20XRC'

    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")

    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
say("[INFO] Loading target position bounding boxes of suspected organoid regions from JSON file")

target_bounding_boxes = load_target_locations(target_filename)
if not target_bounding_boxes:
    say("[ERROR] No organoid regions found in the JSON file. Aborting acquisition.")
else:
    say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of suspected organoid regions")
    first_organoid_x, first_organoid_y, _, _ = target_bounding_boxes[0]
    say(f"[ACTION] Moving to the location of the first organoid area: X={first_organoid_x} μm, Y={first_organoid_y} μm")

    set_brightness(optimal_brightness)
    say(f"[INFO] Halogen lamp brightness adjusted to optimal level: {optimal_brightness}")

    set_z_position(optimal_z)
    say(f"[INFO] Auto-focus completed, Z position set to optimal value: {optimal_z} μm")
    say("[INFO] Analyzing current field of view to obtain Z-stack range for 3D organoid imaging")
    z_max, z_min = z_stack_range()
    z_step_20x = 1.5 # Recommended Z-stack step for 20x objective

    say("[INFO] Configuring automatic image acquisition parameters")
    current_exposure = get_exposure()
    add_channels(channel='1-NONE', exposure=current_exposure)
    say(f"[INFO] Brightfield channel configured with current exposure: {current_exposure} ms")
    for i, (center_x, center_y, width, height) in enumerate(target_bounding_boxes):
        add_acquisition_position(name=f"organoid_{i:03d}", x=center_x, y=center_y, width=width, height=height)
    say(f"[INFO] Configured {len(target_bounding_boxes)} XY positions for suspected organoid regions")
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step_20x)
    say(f"[INFO] Z-stack parameters configured: start={z_min} μm, end={z_max} μm, step={z_step_20x} μm")
    say("[INFO] Initiating automated image acquisition with configured parameters")
    run_acquisition()
    say("[INFO] Automated image acquisition of organoid regions completed")
