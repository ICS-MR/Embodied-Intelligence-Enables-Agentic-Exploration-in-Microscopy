fiji_initialize()

brightfield_file = "brightfield_zstack.ome.tif"
dapi_file = "dapi_fluorescence_zstack.ome.tif"
fitc_file = "fitc_fluorescence_zstack.ome.tif"
tritc_file = "tritc_fluorescence_zstack.ome.tif"

say("[ACTION] Loading brightfield Z-stack image: " + brightfield_file)
brightfield_zstack = load_image(brightfield_file)

say("[ACTION] Loading DAPI fluorescence Z-stack image: " + dapi_file)
dapi_zstack = load_image(dapi_file)

say("[ACTION] Loading FITC fluorescence Z-stack image: " + fitc_file)
fitc_zstack = load_image(fitc_file)

say("[ACTION] Loading TRITC fluorescence Z-stack image: " + tritc_file)
tritc_zstack = load_image(tritc_file)

say("[ACTION] Performing extended depth of field processing on brightfield Z-stack via maximum intensity Z-projection")
brightfield_edf = z_projection(brightfield_zstack, method="max")

say("[ACTION] Performing extended depth of field processing on DAPI Z-stack via maximum intensity Z-projection")
dapi_edf = z_projection(dapi_zstack, method="max")

say("[ACTION] Performing extended depth of field processing on FITC Z-stack via maximum intensity Z-projection")
fitc_edf = z_projection(fitc_zstack, method="max")

say("[ACTION] Performing extended depth of field processing on TRITC Z-stack via maximum intensity Z-projection")
tritc_edf = z_projection(tritc_zstack, method="max")

merged_channels = [brightfield_edf, dapi_edf, fitc_edf, tritc_edf]
channel_colors = ["Brightfield", "Blue", "Green", "Red"]
output_file = "merged_brightfield_dapi_fitc_tritc_rgb.ome.tif"
merged_image = merge_channels(merged_channels, colors=channel_colors, outpath=output_file)

say("[ACTION] Saving merged RGB overlay image to: " + output_file)
save_image(merged_image, output_file, "Merged RGB overlay of brightfield, DAPI, FITC, and TRITC extended-depth-of-field images")
