say("[ACTION] Initializing Cellpose model for DAPI nucleus segmentation with GPU acceleration")
cellpose_initialize(model_type="cpsam", gpu=True)

say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format from TCZYX")
image_2d = image[0, 0, 0, :, :]  # T=0, C=0, Z=0, HxW

say("[ACTION] Running DAPI nucleus segmentation")
masks = segment(
    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} nuclei".format(np.max(masks)))

say("[ACTION] Saving DAPI nucleus segmentation masks")
mask_filename = "dapi_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="DAPI nucleus segmentation masks from dapi_fluorescence_4x image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))
