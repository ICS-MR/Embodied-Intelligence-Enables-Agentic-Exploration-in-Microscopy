if current_channel != target_channel_dapi:
    set_channel(target_channel_dapi)

if organoid_bounding_boxes:
    first_center_x, first_center_y, first_width, first_height = organoid_bounding_boxes[0]
    say(f"[ACTION] Moving to first organoid position: X={first_center_x} μm, Y={first_center_y} μm")
    set_x_y_position(first_center_x, first_center_y)
    say("[INFO] Performing autofocus on first organoid")

    z_max, z_min = z_stack_range()

    z_min, z_max, z_step = 4155.0, 4170.0, 1.5
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)
