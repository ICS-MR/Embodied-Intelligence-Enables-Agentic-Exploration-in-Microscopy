say("[ACTION] Automatically adjusting halogen lamp brightness.")
brightness = perform_autobrightness()
set_brightness(brightness)
say(f"[INFO] Halogen lamp brightness set to {brightness}.")

say("[ACTION] Performing autofocus via images.")
optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[INFO] Autofocus complete. Z-axis set to {optimal_z} μm.")

say("[ACTION] Obtaining the current Z-axis focusing position of the stage.")
current_z = get_z_position()
say(f"[INFO] Current Z-axis position: {current_z} μm.")

z_positions = [0.5 * current_z, 1.0 * current_z, 1.5 * current_z]
images = []

for idx, z in enumerate(z_positions):
    say(f"[ACTION] Setting Z-axis focusing position to {z} μm.")
    if z < 0:
        z = 0
        say("[INFO] Z position below 0, set to 0 μm.")
    elif z > 10000:
        z = 10000
        say("[INFO] Z position above 10000, set to 10000 μm.")
    set_z_position(z)
    say(f"[ACTION] Acquiring image at current XY position ({get_x_y_position()}) with Brightfield, 20x.")
    add_acquisition_position(f"brightfield_20x_z{idx+1}", *get_x_y_position(), None, None)
    add_channels('1-NONE', get_exposure())
    result = run_acquisition()
    images.append(result[0])

say("[INFO] All images acquired at specified Z positions.")
