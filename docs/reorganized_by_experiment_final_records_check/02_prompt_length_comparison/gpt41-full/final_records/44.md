say("[INFO] Setting objective lens to 20x")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens switched to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")

say("[INFO] Loading detected 2D cell positions from JSON file")
target_filename = "2Dcell_locations_list.json"
say(f"[ACTION] Loading target locations from file {target_filename}")
target_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of detected 2D cells")

say("[INFO] Selecting the position with the largest 2D cell area")
max_area = -1
max_bbox = None
for bbox in target_bounding_boxes:
    _, _, width, height = bbox
    area = width * height
    if area > max_area:
        max_area = area
        max_bbox = bbox
if max_bbox is None:
    say("[ERROR] No bounding boxes found in the loaded target positions")
else:
    center_x, center_y, width, height = max_bbox
    say(f"[ACTION] Moving stage to largest cell area at (X={center_x}, Y={center_y}) μm")
    set_x_y_position(center_x, center_y)

    say("[INFO] Performing automatic brightness adjustment for halogen lamp")
    optimal_brightness = perform_autobrightness()
    set_brightness(optimal_brightness)
    say(f"[INFO] Brightness automatically adjusted to {optimal_brightness}")

    say("[INFO] Performing autofocus on current field of view")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Autofocus completed, optimal Z position: {optimal_z} μm")

    say("[INFO] Configuring image acquisition parameters for 20x field of view")
    current_x, current_y = get_x_y_position()
    add_acquisition_position(name="largest_2D_cell_20x", x=current_x, y=current_y, width=None, height=None)
    target_channel = '1-NONE'
    current_exposure = get_exposure()
    add_channels(channel=target_channel, exposure=current_exposure)
    say(f"[INFO] Acquisition parameters set: position ({current_x}, {current_y}), channel {target_channel}, exposure {current_exposure} ms")

    say("[INFO] Initiating automated image acquisition at largest 2D cell area with 20x objective")
    run_acquisition()
    say("[INFO] Automated image acquisition completed at largest 2D cell area with 20x objective")
