    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} labeled cells".format(np.max(masks)))

mask_filename = "labeled_cell_segmentation_1.tif"
