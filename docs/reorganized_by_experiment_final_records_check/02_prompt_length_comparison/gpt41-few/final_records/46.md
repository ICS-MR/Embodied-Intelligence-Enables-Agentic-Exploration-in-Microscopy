say("[INFO] Adjusting halogen lamp brightness for brightfield mode")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Halogen lamp brightness set to optimal value: {optimal_brightness}")
say("[INFO] Performing autofocus on current field of view")
optimal_z = perform_autofocus()
set_z_position(optimal_z)

say("[INFO] Retrieving current Z-axis focusing position")
current_z = get_z_position()
say(f"[INFO] Current Z-axis position: {current_z} μm")
say("[INFO] Configuring automated image acquisition parameters for brightfield Z-stack")
target_channel = '1-NONE'
current_exposure = get_exposure()
current_x, current_y = get_x_y_position()
add_acquisition_position(name="brightfield_zstack", x=current_x, y=current_y, width=None, height=None)
add_channels(channel=target_channel, exposure=current_exposure)
z_positions = [0.5 * current_z, 1.0 * current_z, 1.5 * current_z]
z_positions = [z for z in z_positions if 0 <= z <= 10000]
if len(z_positions) < 2:
    say("[ERROR] Not enough valid Z positions within hardware limits for Z-stack acquisition")
else:
    z_positions = sorted(z_positions)
    z_start = z_positions[0]
    z_end = z_positions[-1]
    if len(z_positions) == 2:
        z_step = z_end - z_start
    else:
        z_step = z_positions[1] - z_positions[0]
    if z_step <= 0:
        z_step = 1.0
    set_z_stack(z_start=z_start, z_end=z_end, z_step=z_step)
    say(f"[INFO] Z-stack parameters set: start {z_start} μm, end {z_end} μm, step {z_step} μm")
    say("[INFO] Initiating automated image acquisition")
    run_acquisition()
    say("[INFO] Automated image acquisition task completed")
