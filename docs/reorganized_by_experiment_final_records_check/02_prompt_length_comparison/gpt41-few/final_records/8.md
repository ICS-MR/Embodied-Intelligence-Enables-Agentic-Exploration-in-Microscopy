say("[INFO] Starting parameter setup for 96-well plate imaging at 20x in brightfield mode")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
target_channel = '1-NONE'
current_channel = get_channel()
if current_channel != target_channel:
    set_channel(target_channel)
    say(f"[INFO] Filter set to brightfield mode (channel: {target_channel})")
else:
    say(f"[INFO] Filter is already in brightfield mode (channel: {target_channel}), no change needed")
say("[ACTION] Generating positions for each well in 96-well plate")
wells_positions = create_96_wells_positions()
say("[INFO] Performing automatic brightness adjustment for brightfield")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)
say(f"[INFO] Halogen lamp brightness automatically adjusted to {optimal_brightness}")
acquisition_positions = []
z_stack_params = []
for idx, (well_x, well_y) in enumerate(wells_positions):
    say(f"[INFO] Moving to well {idx+1} at position ({well_x}, {well_y}) μm for autofocus and Z-stack range analysis")
    set_x_y_position(well_x, well_y)
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Autofocus completed for well {idx+1}, optimal Z position: {optimal_z} μm")
    z_max, z_min = z_stack_range()
    z_step = 1.5
    z_stack_params.append((z_min, z_max, z_step))
    pos_name = f"well_{idx+1:02d}"
    acquisition_positions.append((pos_name, well_x, well_y, None, None))
say("[INFO] Configuring automated acquisition parameters for all wells")
for i, (pos_name, well_x, well_y, width, height) in enumerate(acquisition_positions):
    add_acquisition_position(name=pos_name, x=well_x, y=well_y, width=width, height=height)
    z_min, z_max, z_step = z_stack_params[i]
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)
    add_channels(channel=target_channel, exposure=get_exposure())
say("[INFO] Setting up time-lapse imaging: 24 frames, 1 hour interval")
set_time_series(num_frames=24, interval_sec=3600)
say("[INFO] Initiating automated image acquisition for 96-well plate time-lapse Z-stack imaging")
run_acquisition()
say("[INFO] Automated image acquisition for 96-well plate completed")
