say("[ACTION] Initializing Cellpose model for cell segmentation with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading brightfield image of the entire culture dish from context")
image_path = 'culture_dish_brightfield.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format (extracting first channel)")
# TCZYX -> select T=0, Z=0, C=0 (grayscale), then HxW
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Running cell segmentation on the brightfield image")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for grayscale
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Saving cell segmentation masks")
mask_filename = "cell_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from culture_dish_brightfield image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing cell area distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False
)

say("[ACTION] Saving cell area analysis results")
analysis_filename = "cell_analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
