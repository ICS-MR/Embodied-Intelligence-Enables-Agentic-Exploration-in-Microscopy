say("[ACTION] Initializing Cellpose model for nucleus segmentation with GPU acceleration")
cellpose_initialize(model_type="nuclei")

say("[ACTION] Reading DAPI fluorescent image from context")
image_path = 'dapi_fluorescence_20x.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Extracting DAPI fluorescent channel (C=0) from TCZYX data")
# TCZYX -> select T=0, Z=0, C=0 (DAPI channel), then HxW
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Running nucleus segmentation on DAPI channel")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Only nucleus channel
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} nuclei".format(np.max(masks)))

mask_filename = "nucleus_segmentation_dapi.tif"
say("[ACTION] Saving nucleus segmentation masks")
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Nucleus segmentation masks from DAPI fluorescent image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing nucleus area distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False
)

analysis_filename = "nucleus_analysis_dapi.csv"
say("[ACTION] Saving nucleus area analysis results")
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
