target_objective = '3-LUCPLFLN20XRC'

    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")

say("[ACTION] Loading detected organoid positions from JSON file")

say("[ACTION] Moving to the first organoid position")
first_organoid_x, first_organoid_y, width, height = organoid_positions[0]
set_x_y_position(first_organoid_x, first_organoid_y)

optimal_z = perform_autofocus()

z_max, z_min = z_stack_range()
