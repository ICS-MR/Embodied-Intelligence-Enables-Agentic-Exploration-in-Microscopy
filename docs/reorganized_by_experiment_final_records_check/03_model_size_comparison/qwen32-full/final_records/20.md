cellpose_initialize(model_type="cyto", gpu=True)

image_path = 'brightfield_4x_zstack.ome.tif'
image = cellpose_read(image_path)

# Assuming T=0, C=0, and using full Z dimension
z_stack_image = image[0, 0, :, :, :]

say("[ACTION] Running 3D cell segmentation on Z-stack")
masks_3d = segment(
    image=z_stack_image,
    channels=[0, 0],  # Brightfield single channel
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] 3D Segmentation completed. Detected {} cells".format(np.max(masks_3d)))

say("[ACTION] Saving 3D cell segmentation masks")
mask_filename = "3d_cell_segmentation_1.tif"
save_masks(
    masks=masks_3d,
    filename=mask_filename,
    description="3D cell segmentation masks from brightfield Z-stack of 2×2 mm area"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

analysis_filename = "cell_quantity_analysis_1.csv"
