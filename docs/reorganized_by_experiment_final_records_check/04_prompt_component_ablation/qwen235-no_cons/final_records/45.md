cell_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(cell_bounding_boxes)} 2D cell bounding boxes")

say("[INFO] Sorting cell positions by area in descending order")
# Sort the bounding boxes by area (width * height) from large to small
sorted_cell_boxes = sorted(cell_bounding_boxes, key=lambda x: x[2]*x[3], reverse=True)
say(f"[INFO] Sorted {len(sorted_cell_boxes)} cell positions by area")

say("[INFO] Configuring brightfield image acquisition parameters for 2D cells")
target_channel = '1-NONE'
current_exposure = get_exposure()
say(f"[INFO] Current exposure time: {current_exposure} ms")

for i, (center_x, center_y, width, height) in enumerate(sorted_cell_boxes):
    position_name = f"cell_{i+1:03d}"
    say(f"[ACTION] Configuring acquisition position: {position_name} at ({center_x}, {center_y})")
    add_acquisition_position(name=position_name, x=center_x, y=center_y, width=width, height=height)

say(f"[INFO] Configured {len(sorted_cell_boxes)} acquisition positions for individual cells")
