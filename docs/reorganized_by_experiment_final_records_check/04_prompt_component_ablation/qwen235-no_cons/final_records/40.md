say("[INFO] Adjusting light source brightness for brightfield imaging")
optimal_brightness = perform_autobrightness()
set_brightness(optimal_brightness)

optimal_z = perform_autofocus()
set_z_position(optimal_z)
say(f"[ACTION] Auto-focus completed, Z position set to {optimal_z} μm")

say("[INFO] Configuring acquisition parameters for full 2D section scan")
current_x, current_y = get_x_y_position()

# Assuming a typical full section size of 3cm x 3cm for a tissue section
section_width = 30000  # 3cm in micrometers

position_name = "full_section_scan"

run_acquisition()
say("[ACTION] Full 2D section scan completed successfully")
