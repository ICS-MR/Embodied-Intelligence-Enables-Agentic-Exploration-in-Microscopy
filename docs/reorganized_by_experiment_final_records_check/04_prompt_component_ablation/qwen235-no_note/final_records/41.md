Therefore, the outcome cannot be judged as fully successful.

    colors=["Brightfield", "Blue", "Green", "Red"],
    outpath="merged_organoids_rgb.tif"

say("[ACTION] Saving RGB superimposed image with fluorescence and brightfield")
save_image(merged_image, "merged_organoids_rgb.tif", "RGB superimposed image with brightfield and fluorescence channels (DAPI, FITC, TRITC)")
