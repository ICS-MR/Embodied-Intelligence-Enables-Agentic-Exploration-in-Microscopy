say("[ACTION] Initializing Cellpose model for cell identification with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading full culture dish image from context")
image_path = 'full_dish_view.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format")
image_2d = image[0, 0, 0, :, :]  # Extract HxW from TCZYX

say("[ACTION] Running cell segmentation inference")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Assuming grayscale image for cell identification
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Saving cell segmentation masks")
mask_filename = "cell_segmentation_masks.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from full culture dish image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing cell count in culture dish")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=1,
    plot=False
)
total_cells = len(analysis_df[analysis_df['bin_idx'] >= 0])

say("[ACTION] Saving cell count analysis results")
analysis_filename = "cell_count_analysis.csv"
analysis_df.to_csv(analysis_filename, index=False)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Total number of cells detected: {}".format(total_cells))
say("[INFO] Workflow completed successfully")
