fiji_initialize()
brightfield_file = "brightfield_20x.ome.tif"
fic_fluo_file = "fic_fluorescence_20x.ome.tif"
tritc_fluo_file = "tritc_fluorescence_20x.ome.tif"

say("[ACTION] Loading brightfield image: " + brightfield_file)
brightfield_image = load_image(brightfield_file)

say("[ACTION] Loading FITC fluorescence image: " + fic_fluo_file)
fic_fluorescence_image = load_image(fic_fluo_file)

say("[ACTION] Loading TRITC fluorescence image: " + tritc_fluo_file)
tritc_fluorescence_image = load_image(tritc_fluo_file)

say("[ACTION] Merging brightfield with FITC and TRITC fluorescence channels")
merged_image = merge_channels(
    [brightfield_image, fic_fluorescence_image, tritc_fluorescence_image],
    colors=["Brightfield", "Green", "Red"],
    outpath="merged_2d_section_overlay.tif"
)

say("[ACTION] Saving merged overlay image")
save_image(merged_image, "merged_2d_section_overlay.tif", "Overlay image with brightfield background and FITC/TRITC fluorescence signals")

fiji_shutdown()
