say("[ACTION] Initializing Cellpose model for labeled cell detection with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading DAPI fluorescence image from context")
image_path = 'dapi_full_section.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format")
image_2d = image[0, 0, 0, :, :]  # Extract HxW from TCZYX

say("[ACTION] Running segmentation on DAPI image to detect labeled cells")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel for nucleus
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} labeled cells".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "labeled_cell_masks.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Segmentation masks for labeled cells in DAPI fluorescence image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing labeled cell count")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False
)

say("[ACTION] Saving labeled cell analysis results")
analysis_filename = "labeled_cell_analysis.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
