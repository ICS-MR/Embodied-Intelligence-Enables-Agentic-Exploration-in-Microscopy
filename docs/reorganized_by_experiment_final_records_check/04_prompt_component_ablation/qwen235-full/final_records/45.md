say("[INFO] Starting to load target bounding boxes of 2D cell regions from JSON file")
target_filename = "2Dcell_locations_list.json"
say(f"[ACTION] Loading target locations from file {target_filename}")
target_bounding_boxes = load_target_locations(target_filename)
if not target_bounding_boxes:
    say("[ERROR] No target bounding boxes loaded, exiting")
else:
    say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of 2D cell regions")
    say("[INFO] Sorting bounding boxes by area in descending order")
    target_bounding_boxes.sort(key=lambda x: x[2] * x[3], reverse=True)
    say(f"[INFO] Bounding boxes sorted by area descending")

    say("[INFO] Moving to the location of the largest 2D cell region")
    largest_box = target_bounding_boxes[0]
    center_x, center_y, width, height = largest_box
    target_x = center_x
    target_y = center_y
    current_x, current_y = get_x_y_position()
    if abs(current_x - target_x) > 1 or abs(current_y - target_y) > 1:
        set_x_y_position(target_x, target_y)
        say(f"[ACTION] Moved to largest 2D cell region at ({target_x}, {target_y}) μm")
    else:
        say(f"[INFO] Already at position of largest 2D cell region: ({current_x}, {current_y}) μm")

    say("[INFO] Performing autofocus on selected 2D cell region")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Autofocus completed, Z position set to {optimal_z} μm")

    say("[INFO] Configuring brightfield image acquisition parameters")
    current_exposure = get_exposure()
    say(f"[INFO] Current exposure time: {current_exposure} ms")
    current_channel = get_channel()
    if current_channel != '1-NONE':
        set_channel('1-NONE')
        say("[INFO] Channel set to brightfield mode")
    say("[INFO] Adding acquisition positions for all 2D cell regions")
    for idx, (center_x, center_y, width, height) in enumerate(target_bounding_boxes):
        position_name = f"cell_region_{idx+1}"
        add_acquisition_position(name=position_name, x=center_x, y=center_y, width=width, height=height)
    say(f"[INFO] Added {len(target_bounding_boxes)} positions for 2D cell regions")
    say(f"[INFO] Configuring brightfield channel with current exposure time: {current_exposure} ms")
    add_channels(channel='1-NONE', exposure=current_exposure)

    say("[INFO] Initiating automated image acquisition for all 2D cell regions")
    run_acquisition()
    say("[INFO] Automated image acquisition completed for all 2D cell regions in descending order of area")
