```python
fiji_initialize()
input_file = "current_field_of_view.ome.tif"
say("[ACTION] Loading captured brightfield image of 2D section: " + input_file)
image = load_image(input_file)

say("[ACTION] Detecting 2D cell regions in the brightfield image")
detection_result = analysis_platform_find_target_positions(image, "2Dcell", "Detected 2D cell regions in brightfield image")

say("[ACTION] Overlaying bounding boxes on detected 2D cell areas in original image")
image_np = convert_to_numpy(image)
image_with_boxes = cv.cvtColor(image_np, cv.COLOR_GRAY2BGR)  # Convert to 3-channel for color drawing

# Draw bounding boxes on the image
for (center_x, center_y, width, height) in detection_result:
    top_left = (int(center_x - width / 2), int(center_y - height / 2))
    bottom_right = (int(center_x + width / 2), int(center_y + height / 2))
    cv.rectangle(image_with_boxes, top_left, bottom_right, (0, 255, 0), 2)  # Green box with thickness 2

# Display the annotated image using matplotlib for 5 seconds
say("[ACTION] Displaying annotated image using plt for 5 seconds")
plt.imshow(cv.cvtColor(image_with_boxes, cv.COLOR_BGR2RGB))  # Convert to RGB for proper display
plt.title("Detected 2D Cell Areas with Bounding Boxes")
plt.axis('off')  # Hide axis
plt.show(block=False)
plt.pause(5)
plt.close()

fiji_shutdown()
```
