say("[ACTION] Initializing Cellpose model for nucleus segmentation with GPU acceleration")
cellpose_initialize(gpu=True, model_type="cpsam")

image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format (extracting HxW from TCZYX)")
# Assuming DAPI is the first (and likely only) channel, and we need the first timepoint and z-slice.
# The description indicates 'channel_names: [(0, 0, 255)]', implying a single channel.
image_2d = image[0, 0, 0, :, :] # T=0, C=0, Z=0, then HxW
say("[INFO] Extracted 2D image with shape: {}".format(image_2d.shape))

say("[ACTION] Executing cell segmentation inference on the DAPI image")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # DAPI is a single channel, segmenting nuclei
    diameter=None,    # Auto-estimate diameter
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True      # Denoise for better segmentation
)
say("[INFO] Segmentation completed. Detected {} labeled cells.".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "dapi_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Labeled cell segmentation masks from DAPI_full_section_scan image"
