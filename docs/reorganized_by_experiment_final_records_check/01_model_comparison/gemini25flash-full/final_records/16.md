say("[INFO] Starting automated imaging workflow for organoid regions")

say("[INFO] Setting objective lens to 20x")
target_objective = '3-LUCPLFLN20XRC'
current_objective = get_objective()
if current_objective != target_objective:
    set_objective(target_objective)
    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")
else:
    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")

say("[INFO] Loading target position bounding boxes of suspected organoid regions from JSON file")
target_filename = "organoid_locations_list.json"
say(f"[ACTION] Loading target locations from file {target_filename}")
target_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(target_bounding_boxes)} bounding boxes of suspected organoid regions")

if not target_bounding_boxes:
    say("[ERROR] No organoid regions found in the JSON file. Aborting acquisition.")
else:
    first_organoid_x, first_organoid_y, _, _ = target_bounding_boxes[0]
    say(f"[ACTION] Moving to the location of the first detected organoid: X={first_organoid_x} μm, Y={first_organoid_y} μm")
    set_x_y_position(first_organoid_x, first_organoid_y)

    say("[INFO] Adjusting light source brightness for brightfield imaging")
    optimal_brightness = perform_autobrightness()
    set_brightness(optimal_brightness)
    say(f"[INFO] Halogen lamp brightness set to optimal value: {optimal_brightness}")

    say("[INFO] Performing auto-focus on the current field of view")
    optimal_z = perform_autofocus()
    set_z_position(optimal_z)
    say(f"[INFO] Auto-focus completed, Z position set to optimal value: {optimal_z} μm")

    say("[INFO] Analyzing current field of view to determine appropriate Z-stack range for 3D imaging")
    z_max, z_min = z_stack_range()
    # Recommended Z-stack sampling step for 20x objective ('3-LUCPLFLN20XRC') is 1.5 μm
    z_step = 1.5
    say(f"[INFO] Recommended Z-stack range: {z_min} μm to {z_max} μm with step {z_step} μm")

    say("[INFO] Configuring automatic image acquisition parameters")

    say("[INFO] Configuring channels and exposure parameters")
    # Brightfield channel
    brightfield_channel = '1-NONE'
    brightfield_exposure = 10.0 # Low exposure for brightfield
    add_channels(channel=brightfield_channel, exposure=brightfield_exposure)
    say(f"[INFO] Brightfield channel '{brightfield_channel}' configured with exposure {brightfield_exposure} ms")

    # FITC fluorescence channel
    fitc_channel = '3-U-FBNA'
    fitc_exposure = 200.0 # High exposure for FITC fluorescence
    add_channels(channel=fitc_channel, exposure=fitc_exposure)
    say(f"[INFO] FITC fluorescence channel '{fitc_channel}' configured with exposure {fitc_exposure} ms")

    say("[INFO] Configuring XY position parameters for all detected organoid regions")
    for i, (center_x, center_y, width, height) in enumerate(target_bounding_boxes):
        position_name = f"organoid_{i+1}"
        add_acquisition_position(name=position_name, x=center_x, y=center_y, width=width, height=height)
        say(f"[INFO] Added acquisition position '{position_name}' at ({center_x}, {center_y}) μm with size {width}x{height} μm")

    say("[INFO] Setting Z-stack parameters")
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)
    say(f"[INFO] Z-stack parameters set: start={z_min} μm, end={z_max} μm, step={z_step} μm")

    say("[INFO] Configuring time series parameters for 24 hours with 30-minute intervals")
    num_frames = int(24 * (60 / 30)) # 24 hours / 0.5 hours per frame = 48 frames
    interval_sec = 30 * 60 # 30 minutes in seconds
    set_time_series(num_frames=num_frames, interval_sec=interval_sec)
    say(f"[INFO] Time series configured: {num_frames} frames, {interval_sec} seconds interval (30 minutes)")

    say("[INFO] Initiating automated image acquisition for all organoids")
    run_acquisition()
    say("[INFO] Automated image acquisition completed successfully")
