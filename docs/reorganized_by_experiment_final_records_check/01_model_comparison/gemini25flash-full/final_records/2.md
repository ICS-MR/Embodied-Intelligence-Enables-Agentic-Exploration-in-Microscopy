say("[INFO] Retrieving coordinates of all wells in the 96-well plate")

z_max, z_min = z_stack_range()
objective_label = get_objective()
if objective_label == '1-UPLFLN4XPH':
    z_step = 7.5
elif objective_label == '2-SOB':
    z_step = 3.0
elif objective_label == '3-LUCPLFLN20XRC':
    z_step = 1.5
elif objective_label == '4-LUCPLFLN40X':

elif objective_label == '5-LUCPLFLN60X':
