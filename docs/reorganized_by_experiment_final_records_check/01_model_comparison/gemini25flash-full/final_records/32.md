```python
say("[ACTION] Initializing Cellpose model for cell counting with GPU acceleration")
cellpose_initialize(gpu=True, model_type="cpsam")

say("[ACTION] Reading image data of the entire culture dish from context")
image_path = 'culture_dish_overview.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format for segmentation")
# Assuming the image is TCZYX and contains a single grayscale channel (C=0)
image_2d = image[0, 0, 0, :, :] # Extract HxW from T=0, C=0, Z=0
say("[INFO] Extracted 2D image with shape: {}".format(image_2d.shape))

say("[ACTION] Executing cell segmentation inference on the read image")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel, treat as cytoplasm for general cell detection
    diameter=None,    # Automatically estimate cell diameter
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cell instances.".format(np.max(masks)))

say("[ACTION] Saving cell segmentation masks")
mask_filename = "segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from culture_dish_overview image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing segmented cell masks to obtain cell area and quantity distribution")
px_size = 0.32 # Pixel size from context
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False # Do not generate a plot in this script
)
say("[INFO] Cell area analysis completed.")

say("[ACTION] Saving the analysis results of cell area and quantity distribution")
analysis_filename = "analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully. Releasing model resources and cleaning up environment.")
```
