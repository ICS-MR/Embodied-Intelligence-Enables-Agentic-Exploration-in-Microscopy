say("[ACTION] Initializing Cellpose model for cell segmentation with GPU acceleration")
cellpose_initialize(gpu=True, model_type="cpsam")

say("[ACTION] Reading extended depth of field image from context")
image_path = 'extended_depth_3d_cells_2x2mm_brightfield.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format for segmentation")
# Assuming the extended depth of field image, even if stored as OME-TIFF, is effectively 2D (HxW).
# Extract the first plane (T=0, C=0, Z=0) if dimensions are TCZYX.
if image.ndim == 5:
    image_2d = image[0, 0, 0, :, :]
elif image.ndim == 2:
    image_2d = image
else:
    say(f"[ERROR] Unexpected image dimensions: {image.ndim}. Expected 2D or 5D (TCZYX). Attempting to use the first 2D plane.")
    # Fallback for potentially unexpected dimensions, try to get the last two dimensions
    image_2d = image.reshape(-1, image.shape[-2], image.shape[-1])[0]

say("[ACTION] Running cell segmentation inference on the 2D image")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Single channel grayscale image
    diameter=None,     # Auto-estimate diameter
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True       # Apply denoising for brightfield images
)
say("[INFO] Segmentation completed. Detected {} cell instances".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from extended_depth_3d_cells_2x2mm_brightfield image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing segmented cell masks for area distribution")
# Retrieve pixel size from the original 3D image description
pixel_size_um_per_px = 1.62
analysis_df = analyze_masks(
    masks=masks,
    px_size=pixel_size_um_per_px,
    unit="μm2",
    bins=20,
    plot=False # Do not plot, just return DataFrame
)
say("[INFO] Cell area analysis completed.")

say("[ACTION] Saving cell area and quantity distribution analysis results")
analysis_filename = "analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
