image_path = 'current_fov_brightfield.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format for segmentation")
# Assuming the brightfield image is a single channel and we need a 2D slice (H, W)
# Extracting the first time point (T=0), first channel (C=0), first Z-slice (Z=0)
image_2d = image[0, 0, 0, :, :]
say("[INFO] Extracted 2D image slice with shape: {}".format(image_2d.shape))

say("[ACTION] Executing cell segmentation inference")
masks = segment(
    image=image_2d,
    channels=[0, 0],  # Assuming single grayscale channel for brightfield
    diameter=None,    # Auto-estimate cell diameter
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True      # Apply denoising for potentially noisy brightfield images
)
say("[INFO] Segmentation completed. Detected {} cell instances.".format(np.max(masks)))

say("[ACTION] Saving generated segmentation masks")
mask_filename = "segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from current_fov_brightfield image"
)
say("[INFO] Segmentation masks saved as: {}".format(mask_filename))

say("[INFO] Workflow completed successfully")
