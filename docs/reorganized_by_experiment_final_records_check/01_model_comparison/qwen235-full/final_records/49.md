# Convert to numpy array for processing
image_np = convert_to_numpy(image)

# Add scale bar
scale_bar_length_pixels = int(100 / image.pixel_size_x_um)  # 100 micrometers scale bar
cv.rectangle(image_np, (50, image_np.shape[0] - 50), (50 + scale_bar_length_pixels, image_np.shape[0] - 30), (255, 255, 255), -1)

# Add text label
cv.putText(image_np, "3x3 mm Region", (50, image_np.shape[0] - 10), cv.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2)

# Display using matplotlib
plt.figure(figsize=(10, 10))
plt.imshow(image_np, cmap='gray')
plt.title("Brightfield Image with Scale Bar")
plt.axis('off')
plt.show(block=False)
say("[ACTION] Displaying processed image for 10 seconds")

plt.close()
