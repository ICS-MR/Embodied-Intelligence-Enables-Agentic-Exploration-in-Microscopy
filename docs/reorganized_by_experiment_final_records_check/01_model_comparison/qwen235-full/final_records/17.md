    say(f"[INFO] Objective lens set to 20x (label: {target_objective})")

    say(f"[INFO] Objective lens is already 20x (label: {target_objective}), no change needed")
say("[INFO] Loading organoid position bounding boxes from JSON file")
target_filename = "organoid_locations_list.json"

organoid_bounding_boxes = load_target_locations(target_filename)
say(f"[INFO] Successfully loaded {len(organoid_bounding_boxes)} organoid positions")
if len(organoid_bounding_boxes) > 0:
    first_center_x, first_center_y, _, _ = organoid_bounding_boxes[0]
    say(f"[ACTION] Moving to first organoid position: X={first_center_x} μm, Y={first_center_y} μm")
    set_x_y_position(first_center_x, first_center_y)
    say("[INFO] Performing auto-focus on organoid field of view")

    z_max, z_min = z_stack_range()

    z_step = 1.5
    set_z_stack(z_start=z_min, z_end=z_max, z_step=z_step)
    say(f"[INFO] Z-stack parameters set: range {z_min} to {z_max} μm with step {z_step} μm")
    say("[INFO] Configuring fluorescence channels for automated acquisition")

    add_channels(channel='2-U-FUNA', exposure=200.0)

    # TRITC channel

    say("[INFO] Configuring acquisition positions for all detected organoids")
    for i, (center_x, center_y, width, height) in enumerate(organoid_bounding_boxes):
        position_name = f"organoid_{i+1}"
        add_acquisition_position(name=position_name, x=center_x, y=center_y, width=width, height=height)
        say(f"[ACTION] Added acquisition position {i+1}: {position_name} at X={center_x}, Y={center_y}")
    say("[INFO] Initiating automated multi-channel fluorescence image acquisition")
    run_acquisition()
    say("[INFO] Automated multi-channel fluorescence image acquisition completed")
else:
    say("[ERROR] No organoid positions found in the JSON file")
