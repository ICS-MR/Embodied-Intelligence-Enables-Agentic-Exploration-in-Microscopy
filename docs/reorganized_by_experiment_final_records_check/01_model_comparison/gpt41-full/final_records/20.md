say("[ACTION] Initializing Cellpose model for 3D cell analysis with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading extended depth of field image from context")
image_path = 'extended_depth_brightfield_2mm.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Preprocessing image to 2D format for segmentation")
# Assuming the image is TCZYX or CYX, extract 2D plane (if needed)
if image.ndim == 5:
    # TCZYX: select T=0, C=0, Z=0, then HxW
    image_2d = image[0, 0, 0, :, :]
elif image.ndim == 3:
    # CYX: select C=0
    image_2d = image[0, :, :]
else:
    image_2d = image

say("[ACTION] Running cell segmentation on extended depth image")
masks = segment(
    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} cells".format(np.max(masks)))

say("[ACTION] Saving segmentation masks")
mask_filename = "cell_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Cell segmentation masks from extended depth brightfield 2mm image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing cell number distribution in 2x2 mm area")
px_size = 1.62
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,
    unit="μm2",
    bins=20,
    plot=False
)

say("[ACTION] Saving cell number distribution analysis results")
analysis_filename = "cell_number_analysis_1.csv"
save_csv(df=analysis_df, filename=analysis_filename)
say("[INFO] Analysis results saved as: {}".format(analysis_filename))

say("[INFO] Workflow completed successfully")
