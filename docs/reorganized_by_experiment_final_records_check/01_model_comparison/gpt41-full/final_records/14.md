say("[ACTION] Initializing Cellpose model for nucleus segmentation with GPU acceleration")
cellpose_initialize(model_type="cpsam")

say("[ACTION] Reading DAPI (blue fluorescence) image from context")
image_path = 'dapi_20x_3x3mm.ome.tif'
image = cellpose_read(image_path)
say("[INFO] Image loaded with shape: {}".format(image.shape))

say("[ACTION] Extracting 2D image from TCZYX data (T=0, C=0, Z=0)")
image_2d = image[0, 0, 0, :, :]

say("[ACTION] Running nucleus segmentation on DAPI image")
masks = segment(
    image=image_2d,
    channels=[0, 0],
    diameter=None,
    flow_threshold=0.4,
    cellprob_threshold=0.0,
    min_size=15,
    denoise=True
)
say("[INFO] Segmentation completed. Detected {} nuclei".format(np.max(masks)))

say("[ACTION] Saving nucleus segmentation masks")
mask_filename = "nucleus_segmentation_1.tif"
save_masks(
    masks=masks,
    filename=mask_filename,
    description="Nucleus segmentation masks from DAPI 20x 3x3mm image"
)
say("[INFO] Masks saved as: {}".format(mask_filename))

say("[ACTION] Analyzing nucleus area distribution")
px_size = 0.32
analysis_df = analyze_masks(
    masks=masks,
    px_size=px_size,

    plot=False

analysis_filename = "nucleus_analysis_1.csv"
