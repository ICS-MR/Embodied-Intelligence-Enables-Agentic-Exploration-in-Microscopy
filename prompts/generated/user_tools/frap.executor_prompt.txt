# Example output
import os

# --- Step 1: Create and validate a FRAP point plan ---
plan = create_empty_plan(image_width=512, image_height=512)
say("Created empty point plan.")

# Manually inject some target points into the plan for demonstration
plan["targets"] = [
    {"id": 1, "x": 80,  "y": 90,  "label": "cell_A"},
    {"id": 2, "x": 200, "y": 150, "label": "cell_B"},
    {"id": 3, "x": 310, "y": 300, "label": "cell_C"},
    {"id": 4, "x": 5,   "y": 5,   "label": "too_close_to_border"},  # will be rejected
    {"id": 5, "x": 205, "y": 155, "label": "too_close_to_cell_B"},  # will be rejected
]

# Validate: reject border points, enforce minimum spacing, limit to 3 targets, sort by xy
validated_plan = validate_point_plan(
    plan,
    border_margin_px=20,
    min_spacing_px=30,
    max_targets=3,
    sort_mode="xy"
)
say(f"Validated plan contains {len(validated_plan.get('targets', []))} target(s).")

# --- Step 2: Save and reload the point plan ---
plan_path = save_point_plan(validated_plan, filename="my_frap_points.json")
say(f"Point plan saved to: {plan_path}")

loaded_plan = load_point_plan("my_frap_points.json")
say("Point plan reloaded successfully.")

# --- Step 3: Build a linear image-to-screen mapping ---
mapping = build_linear_mapping(
    image_width=512,
    image_height=512,
    screen_left=100,
    screen_top=150,
    screen_width=800,
    screen_height=800,
    flip_x=False,
    flip_y=True,
    mapping_name="microscope_screen"
)
say("Linear mapping created.")

mapping_path = save_mapping(mapping, filename="my_frap_mapping.json")
say(f"Mapping saved to: {mapping_path}")

loaded_mapping = load_mapping("my_frap_mapping.json")
say("Mapping reloaded successfully.")

# --- Step 4: Map the point plan to screen-space click coordinates ---
click_plan = map_point_plan(loaded_plan, loaded_mapping, clamp_to_region=True)
say(f"Click plan generated with {len(click_plan.get('clicks', []))} click(s).")

click_plan_path = save_click_plan(click_plan, filename="my_frap_click_plan.json")
say(f"Click plan saved to: {click_plan_path}")

loaded_click_plan = load_click_plan("my_frap_click_plan.json")
say("Click plan reloaded successfully.")

# --- Step 5: Create and save a GUI profile for the FRAP application ---
ui_profile = create_ui_profile(
    profile_name="FRAPSoft v2",
    window_title_keyword="FRAPSoft",
    image_left=60,
    image_top=80,
    image_width=512,
    image_height=512,
    controls={
        "start_frap": {"type": "click", "offset_x": 700, "offset_y": 50},
        "reset":      {"type": "hotkey", "keys": ["ctrl", "r"]},
    },
    workflow={
        "pre_actions":  ["reset"],
        "post_actions": ["start_frap"],
    },
    activate_before_action=True,
    click_interval_sec=0.2,
    move_duration_sec=0.05,
    flip_x=False,
    flip_y=False,
)
say("UI profile created.")

profile_path = save_ui_profile(ui_profile, filename="my_frap_profile.json")
say(f"UI profile saved to: {profile_path}")

loaded_profile = load_ui_profile("my_frap_profile.json")
say("UI profile reloaded successfully.")

# --- Step 6: Inspect the FRAP application window (dry-run safe) ---
window_info = inspect_window(title_keyword="FRAPSoft")
say(f"Window inspection result: {window_info}")

# --- Step 7: Simulate individual GUI actions (dry_run=True) ---
activation_result = activate_window(title_keyword="FRAPSoft", dry_run=True)
say(f"Activate window (dry run): {activation_result}")

click_result = click_window_relative(
    title_keyword="FRAPSoft",
    offset_x=700, offset_y=50,
    move_duration_sec=0.05,
    dry_run=True
)
say(f"Window-relative click (dry run): {click_result}")

hotkey_result = press_hotkey(keys=["ctrl", "r"], interval_sec=0.05, dry_run=True)
say(f"Hotkey (dry run): {hotkey_result}")

type_result = type_text(text="FRAP_experiment_01", interval_sec=0.02, dry_run=True)
say(f"Type text (dry run): {type_result}")

# --- Step 8: Execute the click plan (dry_run=True) ---
exec_result = execute_click_plan(
    click_plan_path="my_frap_click_plan.json",
    click_interval_sec=0.2,
    activate_window=True,
    window_title_keyword="FRAPSoft",
    move_duration_sec=0.05,
    dry_run=True
)
say(f"Click plan execution (dry run): {exec_result}")

# --- Step 9: Run named UI actions from the profile (dry_run=True) ---
actions_result = run_ui_actions(
    profile_path="my_frap_profile.json",
    action_names=["reset", "start_frap"],
    dry_run=True
)
say(f"UI actions batch (dry run): {actions_result}")

# --- Step 10: Run the full FRAP workflow (dry_run=True) ---

# --- Step 11: Generate a visual preview of the point plan on the source image ---
